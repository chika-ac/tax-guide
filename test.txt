Sub CopyDevToProd()
    Dim wb As Workbook
    Dim basePath As String
    Dim fso As Object
    Dim devFolder As String
    Dim prodFolder As String

    Set wb = ThisWorkbook
    basePath = wb.Path

    Set fso = CreateObject("Scripting.FileSystemObject")
    devFolder  = basePath & "\..\dev"
    prodFolder = basePath & "\..\prod"

    If Not fso.FolderExists(prodFolder) Then fso.CreateFolder prodFolder
    fso.CopyFile devFolder & "\app.js",     prodFolder & "\app.js",     True
    fso.CopyFile devFolder & "\styles.css", prodFolder & "\styles.css", True
    fso.CopyFile devFolder & "\index.html", prodFolder & "\index.html", True

    Set fso = Nothing
    MsgBox "[完了] dev から prod にコピーしました。", vbInformation, "CopyDevToProd"
End Sub
