① HTMLの morningLinks のクラスを変更
消す：
html<div id="morningLinks" class="morning-links"></div>
追加：
html<div id="morningLinks" class="app-grid"></div>

② JS の buildMorningLinks 関数を丸ごと置き換え
消す（この関数全体）：
js(function buildMorningLinks() {
  const container = document.getElementById("morningLinks");
  morningFiles.forEach((f, i) => {
    if (i > 0) {
      const sep = document.createElement("span");
      sep.className = "morning-sep";
      sep.textContent = "・";
      container.appendChild(sep);
    }
    const a = document.createElement("a");
    a.href = f.url;
    a.textContent = f.team + f.label;
    container.appendChild(a);
  });
})();
追加：
js(function buildMorningLinks() {
  const container = document.getElementById("morningLinks");
  morningFiles.forEach(f => {
    const a = document.createElement("a");
    a.href = f.url;
    a.className = "app-card event-card";
    a.innerHTML = `
      <div class="app-icon">${calIcon}</div>
      <span class="app-name">${f.team}${f.label}</span>
    `;
    container.appendChild(a);
  });
})();

③ CSSの .morning-links 関連を削除
消す（このブロック全体）：
css.morning-links { ... }
.morning-links a { ... }
.morning-sep { ... }
これで朝礼リンクがアンバー系のカードボタンになります。
