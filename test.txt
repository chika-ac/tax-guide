' ───────────────────────────────────────
' ① CSV取り込み
' ───────────────────────────────────────
Sub ImportCSV()
    Dim fPath As String
    fPath = Application.GetOpenFilename("CSVファイル,*.csv", , "CSVを選択してください")
    If fPath = "False" Then Exit Sub

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets(DATA_SHEET)
    ws.Cells.ClearContents
    ws.Cells.ClearFormats

    ' ADO読み込み（CP932）
    Dim cn As Object, rs As Object
    Set cn = CreateObject("ADODB.Connection")
    Dim folder As String: folder = Left(fPath, InStrRev(fPath, "\"))
    Dim fname  As String: fname  = Mid(fPath,  InStrRev(fPath, "\") + 1)

    On Error GoTo FallbackRead
    cn.Open "Provider=Microsoft.Jet.OLEDB.4.0;" & _
            "Data Source=" & folder & ";" & _
            "Extended Properties=""text;HDR=YES;FMT=Delimited;CharacterSet=932"""
    Set rs = cn.Execute("SELECT * FROM [" & fname & "]")

    Dim col As Integer
    For col = 0 To rs.Fields.Count - 1
        ws.Cells(1, col + 1).Value = rs.Fields(col).Name
    Next col
    Dim r As Long: r = 2
    Do While Not rs.EOF
        For col = 0 To rs.Fields.Count - 1
            ws.Cells(r, col + 1).Value = rs.Fields(col).Value
        Next col
        r = r + 1
        rs.MoveNext
    Loop
    rs.Close: cn.Close
    GoTo AfterRead

FallbackRead:
    On Error GoTo 0
    ' テキスト読み込みフォールバック
    Dim fso As Object, ts As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.OpenTextFile(fPath, 1, False, -2)
    Dim lineNum As Long: lineNum = 1
    Do While Not ts.AtEndOfStream
        Dim lineText As String: lineText = ts.ReadLine
        Dim cols() As String: cols = SplitCSVLine(lineText)
        Dim i As Integer
        For i = 0 To UBound(cols)
            ws.Cells(lineNum, i + 1).Value = cols(i)
        Next i
        lineNum = lineNum + 1
    Loop
    ts.Close
    r = lineNum

AfterRead:
    ApplyHeaderStyle ws, 1
    ws.Rows(1).AutoFilter
    ws.Columns("A:W").AutoFit
    MsgBox "CSV取り込み完了！  " & (r - 2) & " 件", vbInformation
    Call UpdateAllSheets
End Sub

' CSV行をダブルクォート考慮で分割
Function SplitCSVLine(s As String) As String()
    Dim result() As String
    ReDim result(0)
    Dim inQ As Boolean: inQ = False
    Dim cur As String: cur = ""
    Dim idx As Integer: idx = 0
    Dim i As Integer
    For i = 1 To Len(s)
        Dim ch As String: ch = Mid(s, i, 1)
        If ch = Chr(34) Then
            If inQ And Mid(s, i + 1, 1) = Chr(34) Then
                cur = cur & Chr(34): i = i + 1
            Else
                inQ = Not inQ
            End If
        ElseIf ch = "," And Not inQ Then
            result(idx) = cur
            cur = "": idx = idx + 1
            ReDim Preserve result(idx)
        Else
            cur = cur & ch
        End If
    Next i
    result(idx) = cur
    SplitCSVLine = result
End Function

' ───────────────────────────────────────
' 共通ヘルパー
' ───────────────────────────────────────
Function ColIdx(ws As Worksheet, hdr As String) As Long
    Dim c As Range
    For Each c In ws.Rows(1).Cells
        If Trim(c.Value) = hdr Then ColIdx = c.Column: Exit Function
    Next c
    ColIdx = 0
End Function

Function DateStr(v As Variant) As String
    On Error GoTo bad
    Dim s As String: s = CStr(v)
    If s = "" Or s = "Empty" Then DateStr = "": Exit Function
    If IsDate(s) Then
        DateStr = Format(CDate(s), "M/d"): Exit Function
    End If
    Dim p() As String: p = Split(s, " ")
    Dim d() As String: d = Split(p(0), "/")
    DateStr = CInt(d(1)) & "/" & CInt(d(2))
    Exit Function
bad: DateStr = ""
End Function

Function WeekMon(v As Variant) As Date
    On Error GoTo bad
    Dim d As Date
    Dim s As String: s = CStr(v)
    If IsDate(s) Then
        d = CDate(s)
    Else
        d = CDate(Split(s, " ")(0))
    End If
    WeekMon = d - (Weekday(d, vbMonday) - 1)
    Exit Function
bad: WeekMon = CDate("1900/1/1")
End Function

Sub ApplyHeaderStyle(ws As Worksheet, r As Long)
    Dim lastC As Long: lastC = ws.Cells(r, ws.Columns.Count).End(xlToLeft).Column
    With ws.Range(ws.Cells(r, 1), ws.Cells(r, lastC))
        .Font.Bold = True
        .Font.Name = "Meiryo UI"
        .Font.Color = RGB(255, 255, 255)
        .Interior.Color = RGB(31, 56, 100)
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .WrapText = True
        .RowHeight = 28
        .Borders.LineStyle = xlContinuous
        .Borders.Color = RGB(189, 215, 238)
    End With
End Sub

Sub ApplyDataStyle(ws As Worksheet, r1 As Long, r2 As Long, c1 As Long, c2 As Long)
    Dim r As Long
    For r = r1 To r2
        With ws.Range(ws.Cells(r, c1), ws.Cells(r, c2))
            .Interior.Color = IIf(r Mod 2 = 0, RGB(235, 242, 250), RGB(255, 255, 255))
            .Font.Name = "Meiryo UI"
            .Font.Size = 10
            .VerticalAlignment = xlCenter
            .Borders.LineStyle = xlContinuous
            .Borders.Color = RGB(189, 215, 238)
        End With
    Next r
End Sub

Sub ApplyTotalStyle(ws As Worksheet, r As Long, c1 As Long, c2 As Long)
    With ws.Range(ws.Cells(r, c1), ws.Cells(r, c2))
        .Font.Bold = True
        .Interior.Color = RGB(31, 56, 100)
        .Font.Color = RGB(255, 255, 255)
        .Font.Name = "Meiryo UI"
        .Borders.LineStyle = xlContinuous
        .Borders.Color = RGB(255, 255, 255)
    End With
End Sub

Sub AutoFitSheet(ws As Worksheet)
    ws.Columns.AutoFit
    Dim c As Range
    For Each c In ws.UsedRange.Columns
        If c.ColumnWidth > 22 Then c.ColumnWidth = 22
        If c.ColumnWidth < 4  Then c.ColumnWidth = 4
    Next c
    ws.Columns(1).ColumnWidth = 18
End Sub

' ───────────────────────────────────────
' ③ 新規履歴件数シート
' ───────────────────────────────────────
Sub BuildHistorySheet()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_HISTORY)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    If colD = 0 Then Exit Sub

    Dim dc As Object: Set dc = CreateObject("Scripting.Dictionary")
    Dim r As Long
    For r = 2 To lastRow
        Dim ds As String: ds = DateStr(dws.Cells(r, colD).Value)
        If ds <> "" Then
            dc(ds) = IIf(dc.Exists(ds), dc(ds) + 1, 1)
        End If
    Next r

    ws.Cells(1, 1).Value = "新規履歴件数"
    With ws.Cells(1, 1).Font
        .Bold = True: .Size = 13: .Name = "Meiryo UI": .Color = RGB(31, 56, 100)
    End With
    ws.Row_Dimensions(1).RowHeight = 24

    ws.Cells(2, 1).Value = "区分"
    Dim col As Long: col = 2
    Dim k As Variant
    For Each k In dc.Keys
        ws.Cells(2, col).Value = k: col = col + 1
    Next k
    ws.Cells(2, col).Value = "合計"
    ApplyHeaderStyle ws, 2

    ws.Cells(3, 1).Value = "件数"
    col = 2
    For Each k In dc.Keys
        ws.Cells(3, col).Value = dc(k): col = col + 1
    Next k
    ws.Cells(3, col).Value = "=SUM(" & ws.Cells(3, 2).Address(False, False) & ":" & ws.Cells(3, col - 1).Address(False, False) & ")"
    ApplyDataStyle ws, 3, 3, 1, col
    With ws.Cells(3, 1)
        .Interior.Color = RGB(214, 228, 247): .Font.Bold = True
    End With
    AutoFitSheet ws
End Sub

' ───────────────────────────────────────
' ④ カテゴリー別詳細シート
' ───────────────────────────────────────
Sub BuildCatDetailSheet()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_CAT_DTL)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    Dim colC As Long: colC = ColIdx(dws, "業務分類詳細")
    If colD = 0 Or colC = 0 Then Exit Sub

    Dim counts As Object:   Set counts   = CreateObject("Scripting.Dictionary")
    Dim bigSet As Object:   Set bigSet   = CreateObject("Scripting.Dictionary")
    Dim smallSet As Object: Set smallSet = CreateObject("Scripting.Dictionary")
    Dim dateSet As Object:  Set dateSet  = CreateObject("Scripting.Dictionary")

    Dim r As Long
    For r = 2 To lastRow
        Dim ds As String: ds = DateStr(dws.Cells(r, colD).Value)
        Dim cv As String: cv = Trim(CStr(dws.Cells(r, colC).Value))
        If ds = "" Or cv = "" Or cv = "Empty" Then GoTo NextRow1
        Dim p() As String: p = Split(cv, Chr(12288)) '全角スペース
        Dim big As String:   big   = Trim(p(0))
        Dim small As String: small = IIf(UBound(p) >= 1, Trim(p(1)), "（その他）")
        Dim kbs As String:   kbs   = big & Chr(9) & small
        Dim kf  As String:   kf    = kbs & Chr(9) & ds
        If Not bigSet.Exists(big) Then   bigSet.Add big, True
        If Not smallSet.Exists(kbs) Then smallSet.Add kbs, big
        If Not dateSet.Exists(ds) Then   dateSet.Add ds, True
        counts(kf) = IIf(counts.Exists(kf), counts(kf) + 1, 1)
NextRow1:
    Next r

    ws.Cells(1, 1).Value = "カテゴリー別詳細"
    With ws.Cells(1, 1).Font
        .Bold = True: .Size = 13: .Name = "Meiryo UI": .Color = RGB(31, 56, 100)
    End With

    ws.Cells(2, 1).Value = "大分類"
    ws.Cells(2, 2).Value = "小分類"
    Dim col As Long: col = 3
    Dim dk As Variant
    For Each dk In dateSet.Keys
        ws.Cells(2, col).Value = dk: col = col + 1
    Next dk
    Dim tcol As Long: tcol = col
    ws.Cells(2, tcol).Value = "合計"
    ApplyHeaderStyle ws, 2

    Dim rOut As Long: rOut = 3
    Dim bk As Variant
    For Each bk In bigSet.Keys
        Dim firstInBig As Boolean: firstInBig = True
        Dim sk As Variant
        For Each sk In smallSet.Keys
            If smallSet(sk) = bk Then
                Dim pp() As String: pp = Split(sk, Chr(9))
                If firstInBig Then
                    ws.Cells(rOut, 1).Value = pp(0)
                    firstInBig = False
                End If
                ws.Cells(rOut, 2).Value = pp(1)
                col = 3
                For Each dk In dateSet.Keys
                    Dim kf2 As String: kf2 = sk & Chr(9) & dk
                    ws.Cells(rOut, col).Value = IIf(counts.Exists(kf2), counts(kf2), 0)
                    col = col + 1
                Next dk
                ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 3).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
                rOut = rOut + 1
            End If
        Next sk
    Next bk

    ' 合計行
    ws.Cells(rOut, 1).Value = "合計"
    col = 3
    For Each dk In dateSet.Keys
        ws.Cells(rOut, col).Value = "=SUM(" & ws.Cells(3, col).Address(False, False) & ":" & ws.Cells(rOut - 1, col).Address(False, False) & ")"
        col = col + 1
    Next dk
    ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 3).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"

    ApplyDataStyle ws, 3, rOut - 1, 1, tcol
    ApplyTotalStyle ws, rOut, 1, tcol

    ' 大分類セル装飾
    Dim rr As Long
    For rr = 3 To rOut - 1
        If Trim(ws.Cells(rr, 1).Value) <> "" Then
            ws.Cells(rr, 1).Interior.Color = RGB(214, 228, 247)
            ws.Cells(rr, 1).Font.Bold = True
        End If
    Next rr

    ' 大分類セルを縦結合
    Dim ms As Long: ms = 3
    Dim cb As String: cb = ws.Cells(3, 1).Value
    For rr = 4 To rOut
        Dim cv2 As String: cv2 = Trim(ws.Cells(rr, 1).Value)
        If cv2 <> "" And cv2 <> cb Then
            If rr - 1 > ms Then
                ws.Range(ws.Cells(ms, 1), ws.Cells(rr - 1, 1)).Merge
                ws.Cells(ms, 1).VerticalAlignment = xlCenter
            End If
            ms = rr: cb = cv2
        End If
    Next rr

    AutoFitSheet ws
End Sub

' ───────────────────────────────────────
' ⑤ カテゴリー別（週比較）シート
' ───────────────────────────────────────
Sub BuildCatWeekSheet()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_CAT_WEEK)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    Dim colC As Long: colC = ColIdx(dws, "業務分類詳細")
    If colD = 0 Or colC = 0 Then Exit Sub

    Dim counts  As Object: Set counts  = CreateObject("Scripting.Dictionary")
    Dim bigSet  As Object: Set bigSet  = CreateObject("Scripting.Dictionary")
    Dim weekSet As Object: Set weekSet = CreateObject("Scripting.Dictionary") ' key=MM/DD, val=Date

    Dim r As Long
    For r = 2 To lastRow
        Dim wm As Date: wm = WeekMon(dws.Cells(r, colD).Value)
        If wm = CDate("1900/1/1") Then GoTo NextRow5
        Dim wk As String: wk = Format(wm, "MM/DD")
        Dim cv As String: cv = Trim(CStr(dws.Cells(r, colC).Value))
        If cv = "" Or cv = "Empty" Then GoTo NextRow5
        Dim big As String: big = Trim(Split(cv, Chr(12288))(0))
        If Not bigSet.Exists(big)  Then bigSet.Add big, True
        If Not weekSet.Exists(wk)  Then weekSet.Add wk, wm
        Dim kk As String: kk = big & Chr(9) & wk
        counts(kk) = IIf(counts.Exists(kk), counts(kk) + 1, 1)
NextRow5:
    Next r

    ' 週を昇順ソート
    Dim wKeys()  As String: ReDim wKeys(weekSet.Count - 1)
    Dim wDates() As Date:   ReDim wDates(weekSet.Count - 1)
    Dim wi As Long: wi = 0
    Dim wk2 As Variant
    For Each wk2 In weekSet.Keys
        wKeys(wi) = wk2: wDates(wi) = weekSet(wk2): wi = wi + 1
    Next wk2
    Dim i As Long, j As Long
    For i = 0 To UBound(wKeys) - 1
        For j = i + 1 To UBound(wKeys)
            If wDates(j) < wDates(i) Then
                Dim tk As String:  tk = wKeys(i):  wKeys(i) = wKeys(j):   wKeys(j) = tk
                Dim td As Date:    td = wDates(i): wDates(i) = wDates(j): wDates(j) = td
            End If
        Next j
    Next i

    ws.Cells(1, 1).Value = "カテゴリー別（週比較）"
    With ws.Cells(1, 1).Font
        .Bold = True: .Size = 13: .Name = "Meiryo UI": .Color = RGB(31, 56, 100)
    End With

    ws.Cells(2, 1).Value = "大分類"
    Dim cs As Long: cs = 2
    For i = 0 To UBound(wKeys)
        ws.Cells(2, cs).Value = wKeys(i) & "週 今週"
        ws.Cells(2, cs + 1).Value = wKeys(i) & "週 先週"
        ws.Cells(2, cs + 2).Value = wKeys(i) & "週 差分"
        cs = cs + 3
    Next i
    ws.Cells(2, cs).Value = "今週合計"
    ws.Cells(2, cs + 1).Value = "先週合計"
    ws.Cells(2, cs + 2).Value = "差分合計"
    Dim lastHC As Long: lastHC = cs + 2
    ApplyHeaderStyle ws, 2

    Dim rOut As Long: rOut = 3
    Dim bk As Variant
    For Each bk In bigSet.Keys
        ws.Cells(rOut, 1).Value = bk
        cs = 2
        Dim twt As Long: twt = 0
        Dim lwt As Long: lwt = 0
        For i = 0 To UBound(wKeys)
            Dim kk1 As String: kk1 = bk & Chr(9) & wKeys(i)
            Dim tc As Long:    tc  = IIf(counts.Exists(kk1), counts(kk1), 0)
            Dim prevD As Date: prevD = wDates(i) - 7
            Dim pk As String:  pk   = Format(prevD, "MM/DD")
            Dim kk2 As String: kk2  = bk & Chr(9) & pk
            Dim lc As Long:    lc   = IIf(counts.Exists(kk2), counts(kk2), 0)
            Dim df As Long:    df   = tc - lc
            ws.Cells(rOut, cs).Value     = tc
            ws.Cells(rOut, cs + 1).Value = lc
            ws.Cells(rOut, cs + 2).Value = df
            Call ColorDiff(ws.Cells(rOut, cs + 2), df)
            twt = twt + tc: lwt = lwt + lc
            cs = cs + 3
        Next i
        ws.Cells(rOut, cs).Value     = twt
        ws.Cells(rOut, cs + 1).Value = lwt
        Dim tdf As Long: tdf = twt - lwt
        ws.Cells(rOut, cs + 2).Value = tdf
        Call ColorDiff(ws.Cells(rOut, cs + 2), tdf)
        rOut = rOut + 1
    Next bk

    ApplyDataStyle ws, 3, rOut - 1, 1, lastHC

    ' 凡例
    rOut = rOut + 1
    ws.Cells(rOut, 1).Value = "【差分の色凡例】"
    ws.Cells(rOut, 1).Font.Bold = True: ws.Cells(rOut, 1).Font.Name = "Meiryo UI"
    ws.Cells(rOut + 1, 1).Value = "緑：先週比 +5件以上"
    ws.Cells(rOut + 1, 1).Interior.Color = RGB(198, 239, 206)
    ws.Cells(rOut + 2, 1).Value = "黄：先週比 ±4件以内"
    ws.Cells(rOut + 2, 1).Interior.Color = RGB(255, 235, 156)
    ws.Cells(rOut + 3, 1).Value = "赤：先週比 -5件以下"
    ws.Cells(rOut + 3, 1).Interior.Color = RGB(255, 199, 206)

    AutoFitSheet ws
End Sub

Sub ColorDiff(c As Range, df As Long)
    If df >= 5 Then
        c.Interior.Color = RGB(198, 239, 206)
        c.Font.Color     = RGB(0, 97, 0)
        c.Font.Bold      = True
    ElseIf df <= -5 Then
        c.Interior.Color = RGB(255, 199, 206)
        c.Font.Color     = RGB(156, 0, 6)
        c.Font.Bold      = True
    Else
        c.Interior.Color = RGB(255, 235, 156)
        c.Font.Color     = RGB(156, 87, 0)
    End If
End Sub

    For Each dk In dateSet.Keys
        ws.Cells(rOut, col).Value = "=SUM(" & ws.Cells(3, col).Address(False, False) & ":" & ws.Cells(rOut - 1, col).Address(False, False) & ")"
        col = col + 1
    Next dk
    ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 2).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
    ApplyDataStyle ws, 3, rOut - 1, 1, tcol
    ApplyTotalStyle ws, rOut, 1, tcol
    AutoFitSheet ws
End Sub

' ───────────────────────────────────────
' ⑦ 受付担当者別対応件数シート
' ───────────────────────────────────────
Sub BuildStaffSheet()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_STAFF)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD  As Long: colD  = ColIdx(dws, "初回受付日時")
    Dim col1  As Long: col1  = ColIdx(dws, "受付担当①")
    Dim col2  As Long: col2  = ColIdx(dws, "受付担当②")
    Dim col3  As Long: col3  = ColIdx(dws, "受付担当③")
    Dim col4  As Long: col4  = ColIdx(dws, "受付担当④")
    Dim col5  As Long: col5  = ColIdx(dws, "受付担当⑤")
    If colD = 0 Or col1 = 0 Then Exit Sub

    Dim sCols(1 To 5) As Long
    sCols(1) = col1: sCols(2) = col2: sCols(3) = col3: sCols(4) = col4: sCols(5) = col5

    Dim counts   As Object: Set counts   = CreateObject("Scripting.Dictionary") ' staff|date
    Dim numCnts  As Object: Set numCnts  = CreateObject("Scripting.Dictionary") ' staff|num
    Dim staffSet As Object: Set staffSet = CreateObject("Scripting.Dictionary")
    Dim dateSet  As Object: Set dateSet  = CreateObject("Scripting.Dictionary")

    Dim r As Long, sn As Long
    For r = 2 To lastRow
        Dim ds As String: ds = DateStr(dws.Cells(r, colD).Value)
        If ds = "" Then GoTo NextRow7
        For sn = 1 To 5
            If sCols(sn) = 0 Then GoTo NextSN
            Dim sv As String: sv = Trim(CStr(dws.Cells(r, sCols(sn)).Value))
            If sv = "" Or sv = "Empty" Then GoTo NextSN
            If Not staffSet.Exists(sv)  Then staffSet.Add sv, True
            If Not dateSet.Exists(ds)   Then dateSet.Add ds, True
            Dim kk As String: kk = sv & Chr(9) & ds
            counts(kk) = IIf(counts.Exists(kk), counts(kk) + 1, 1)
            Dim knk As String: knk = sv & Chr(9) & CStr(sn)
            numCnts(knk) = IIf(numCnts.Exists(knk), numCnts(knk) + 1, 1)
NextSN:
        Next sn
NextRow7:
    Next r

    ws.Cells(1, 1).Value = "受付担当者別対応件数"
    With ws.Cells(1, 1).Font
        .Bold = True: .Size = 13: .Name = "Meiryo UI": .Color = RGB(31, 56, 100)
    End With

    ws.Cells(2, 1).Value = "担当者名"
    Dim numColors(1 To 5) As Long
    numColors(1) = RGB(31, 56, 100):   numColors(2) = RGB(46, 117, 182)
    numColors(3) = RGB(68, 144, 196):  numColors(4) = RGB(100, 162, 206)
    numColors(5) = RGB(144, 186, 220)
    Dim si As Long
    For si = 1 To 5
        ws.Cells(2, si + 1).Value = "担当" & Choose(si, "①", "②", "③", "④", "⑤") & "件数"
    Next si

    Dim col As Long: col = 7
    Dim dk As Variant
    For Each dk In dateSet.Keys
        ws.Cells(2, col).Value = dk: col = col + 1
    Next dk
    Dim tcol As Long: tcol = col
    ws.Cells(2, tcol).Value = "合計"
    ApplyHeaderStyle ws, 2

    ' 担当番号別ヘッダー色
    For si = 1 To 5
        ws.Cells(2, si + 1).Interior.Color = numColors(si)
    Next si

    Dim rOut As Long: rOut = 3
    Dim sk As Variant
    For Each sk In staffSet.Keys
        ws.Cells(rOut, 1).Value = sk
        For si = 1 To 5
            Dim knk2 As String: knk2 = sk & Chr(9) & CStr(si)
            ws.Cells(rOut, si + 1).Value = IIf(numCnts.Exists(knk2), numCnts(knk2), 0)
        Next si
        col = 7
        For Each dk In dateSet.Keys
            Dim kk2 As String: kk2 = sk & Chr(9) & dk
            ws.Cells(rOut, col).Value = IIf(counts.Exists(kk2), counts(kk2), 0)
            col = col + 1
        Next dk
        ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 7).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
        rOut = rOut + 1
    Next sk

    ' 合計行
    ws.Cells(rOut, 1).Value = "合計"
    For si = 1 To 5
        ws.Cells(rOut, si + 1).Value = "=SUM(" & ws.Cells(3, si + 1).Address(False, False) & ":" & ws.Cells(rOut - 1, si + 1).Address(False, False) & ")"
    Next si
    col = 7
    For Each dk In dateSet.Keys
        ws.Cells(rOut, col).Value = "=SUM(" & ws.Cells(3, col).Address(False, False) & ":" & ws.Cells(rOut - 1, col).Address(False, False) & ")"
        col = col + 1
    Next dk
    ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 7).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
    ApplyDataStyle ws, 3, rOut - 1, 1, tcol
    ApplyTotalStyle ws, rOut, 1, tcol
    AutoFitSheet ws
End Sub

' ───────────────────────────────────────
' フィルタ機能
' ───────────────────────────────────────
Sub FilterData()
    Dim ws  As Worksheet: Set ws  = ThisWorkbook.Sheets(DATA_SHEET)
    Dim pws As Worksheet: Set pws = ThisWorkbook.Sheets(PANEL_SHEET)
    Dim kw  As String: kw  = Trim(pws.Cells(13, 3).Value)
    Dim col As String: col = Trim(pws.Cells(14, 3).Value)

    If ws.AutoFilterMode Then ws.AutoFilterMode = False
    ws.Rows(1).AutoFilter
    If kw = "" Then MsgBox "キーワードを入力してください", vbExclamation: Exit Sub
    Dim tc As Long: tc = ColIdx(ws, col)
    If tc = 0 Then MsgBox "列名 [" & col & "] が見つかりません", vbExclamation: Exit Sub
    ws.Range("A1").AutoFilter Field:=tc, Criteria1:="*" & kw & "*"
    ws.Activate
    MsgBox "[" & col & "] で [" & kw & "] を検索しました", vbInformation
End Sub

Sub ClearFilter()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(DATA_SHEET)
    If ws.AutoFilterMode Then ws.AutoFilterMode = False
    ws.Rows(1).AutoFilter
    MsgBox "フィルターをクリアしました", vbInformation
End Sub
