Option Explicit

' ----------------------------------------------------------------
' STEP1: CSVインポート
' ----------------------------------------------------------------
Sub ImportCSV()
    Dim ws As Worksheet
    Dim fPath As Variant
    Dim fNum As Integer
    Dim line As String
    Dim fields() As String
    Dim rowNum As Long
    Dim isFirst As Boolean
    Dim i As Integer
    Dim lastRow As Long

    fPath = Application.GetOpenFilename("CSV," & Chr(42) & ".csv")
    If VarType(fPath) = vbBoolean Then Exit Sub

    Set ws = ThisWorkbook.Sheets("Data")

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow > 2 Then
        ws.Rows("3:" & lastRow).ClearContents
    End If

    fNum = FreeFile
    Open fPath For Input As #fNum
    rowNum = 3
    isFirst = True

    Do While Not EOF(fNum)
        Line Input #fNum, line
        If isFirst Then
            isFirst = False
        Else
            fields = ParseCSVLine(line)
            If UBound(fields) >= 0 Then
                If Trim(fields(0)) <> "" Then
                    For i = 0 To UBound(fields)
                        If i < 23 Then
                            ws.Cells(rowNum, i + 1).Value = fields(i)
                        End If
                    Next i
                    rowNum = rowNum + 1
                End If
            End If
        End If
    Loop

    Close #fNum

    MsgBox rowNum - 3 & " ken no data wo import shimashita."
End Sub

' ----------------------------------------------------------------
' CSVライン解析（ダブルクォート・カンマ対応）
' ----------------------------------------------------------------
Function ParseCSVLine(line As String) As String()
    Dim result() As String
    Dim count As Integer
    Dim i As Long
    Dim c As String
    Dim current As String
    Dim inQuote As Boolean

    count = 0
    ReDim result(0)
    current = ""
    inQuote = False

    For i = 1 To Len(line)
        c = Mid(line, i, 1)
        If c = Chr(34) Then
            If inQuote And i < Len(line) And Mid(line, i + 1, 1) = Chr(34) Then
                current = current & Chr(34)
                i = i + 1
            Else
                inQuote = Not inQuote
            End If
        ElseIf c = Chr(44) And Not inQuote Then
            result(count) = current
            count = count + 1
            ReDim Preserve result(count)
            current = ""
        Else
            current = current & c
        End If
    Next i
    result(count) = current
    ParseCSVLine = result
End Function

' ----------------------------------------------------------------
' STEP2: マスク実行
' ----------------------------------------------------------------
Sub RunMask()
    Dim wsSrc As Worksheet
    Dim wsMask As Worksheet
    Dim lastRow As Long
    Dim lastRow2 As Long
    Dim i As Long
    Dim j As Integer
    Dim cellVal As String
    Dim maskChar As String

    Set wsSrc = ThisWorkbook.Sheets("Data")
    Set wsMask = ThisWorkbook.Sheets("Masked")
    maskChar = "**"

    lastRow = wsSrc.Cells(wsSrc.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox "Error: No data in Data sheet. Please import CSV first."
        Exit Sub
    End If

    lastRow2 = wsMask.Cells(wsMask.Rows.Count, 1).End(xlUp).Row
    If lastRow2 > 2 Then
        wsMask.Rows("3:" & lastRow2).ClearContents
    End If

    Application.ScreenUpdating = False

    For i = 3 To lastRow
        For j = 1 To 23
            cellVal = CStr(wsSrc.Cells(i, j).Value)
            cellVal = MaskText(cellVal, maskChar)
            wsMask.Cells(i, j).Value = cellVal
            wsMask.Cells(i, j).Font.Name = "Arial"
            wsMask.Cells(i, j).Font.Size = 9
        Next j
    Next i

    Application.ScreenUpdating = True

    MsgBox lastRow - 2 & " ken no mask shori ga kanryo shimashita."
End Sub

' ----------------------------------------------------------------
' テキストマスク処理（正規表現）
' ----------------------------------------------------------------
Function MaskText(text As String, maskChar As String) As String
    Dim re As Object
    Dim result As String
    result = text

    Set re = CreateObject("VBScript.RegExp")
    re.Global = True
    re.IgnoreCase = False

    ' 申請番号（12桁）
    re.Pattern = "\b\d{12}\b"
    result = re.Replace(result, maskChar)

    ' 職員番号（7〜8桁）
    re.Pattern = "\b\d{7,8}\b"
    result = re.Replace(result, maskChar)

    ' 氏名＋敬称（様・さま・サマ・さん）
    re.Pattern = "[^\s\n]{1,10}(" & ChrW(27096) & "|" & ChrW(12373) & ChrW(12414) & "|" & ChrW(12469) & ChrW(12510) & "|" & ChrW(12373) & ChrW(12435) & ")"
    result = re.Replace(result, maskChar)

    ' 年収（円・えん）
    re.Pattern = "\d[\d,]*(" & ChrW(20870) & "|" & ChrW(12360) & ChrW(12435) & ")"
    result = re.Replace(result, maskChar)

    ' 年収（¥記号）
    re.Pattern = ChrW(165) & "[\d,]+"
    result = re.Replace(result, maskChar)

    MaskText = result
End Function

' ----------------------------------------------------------------
' STEP3: 会話分割
' ----------------------------------------------------------------
Sub RunSplit()
    Dim wsMask As Worksheet
    Dim wsSplit As Worksheet
    Dim lastRow As Long
    Dim lastRow3 As Long
    Dim i As Long
    Dim outRow As Long
    Dim j As Integer
    Dim memo As String
    Dim parts() As String
    Dim sp1 As String
    Dim sp2 As String
    Dim role1 As String
    Dim role2 As String
    Dim pairNum As Integer
    Dim isRenraku As Boolean
    Dim startIdx As Integer
    Dim lastIdx As Integer
    Dim adjustedLast As Integer
    Dim roleOP As String
    Dim roleCL As String

    Set wsMask = ThisWorkbook.Sheets("Masked")
    Set wsSplit = ThisWorkbook.Sheets("Split")

    roleOP = "OP"
    roleCL = "CL"

    lastRow = wsMask.Cells(wsMask.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox "Error: No data in Masked sheet. Please run mask first."
        Exit Sub
    End If

    lastRow3 = wsSplit.Cells(wsSplit.Rows.Count, 1).End(xlUp).Row
    If lastRow3 > 2 Then
        wsSplit.Rows("3:" & lastRow3).ClearContents
    End If

    Application.ScreenUpdating = False
    outRow = 3

    For i = 3 To lastRow
        memo = wsMask.Cells(i, 12).Value
        If Trim(memo) = "" Then GoTo NextRow

        ' 改行+↓+改行 で分割、なければ↓のみで分割
        If InStr(memo, Chr(10) & Chr(8595) & Chr(10)) > 0 Then
            parts = Split(memo, Chr(10) & Chr(8595) & Chr(10))
        Else
            parts = Split(memo, Chr(8595))
        End If

        ' 連携発信チェック（最初の発言にあればOPが先）
        isRenraku = (InStr(parts(0), ChrW(36899) & ChrW(25345) & ChrW(30330) & ChrW(20449)) > 0)

        pairNum = 1

        For j = 0 To UBound(parts) - 1

            ' 連携発信ありの場合、最初のペア(j=0)はスキップ
            If isRenraku And j = 0 Then GoTo NextPair

            sp1 = Trim(parts(j))
            sp2 = Trim(parts(j + 1))

            ' 発言者判定
            If isRenraku Then
                ' OP→CL→OP→CL...（j=0がOP）
                If j Mod 2 = 0 Then
                    role1 = roleOP
                    role2 = roleCL
                Else
                    role1 = roleCL
                    role2 = roleOP
                End If
            Else
                ' CL→OP→CL→OP...（j=0がCL）
                If j Mod 2 = 0 Then
                    role1 = roleCL
                    role2 = roleOP
                Else
                    role1 = roleOP
                    role2 = roleCL
                End If
            End If

            wsSplit.Cells(outRow, 1).Value = wsMask.Cells(i, 1).Value
            wsSplit.Cells(outRow, 2).Value = wsMask.Cells(i, 10).Value
            wsSplit.Cells(outRow, 3).Value = pairNum
            wsSplit.Cells(outRow, 4).Value = role1
            wsSplit.Cells(outRow, 5).Value = sp1
            wsSplit.Cells(outRow, 6).Value = role2
            wsSplit.Cells(outRow, 7).Value = sp2
            wsSplit.Rows(outRow).Font.Name = "Arial"
            wsSplit.Rows(outRow).Font.Size = 9
            wsSplit.Cells(outRow, 5).WrapText = True
            wsSplit.Cells(outRow, 7).WrapText = True
            wsSplit.Rows(outRow).RowHeight = 45

            outRow = outRow + 1
            pairNum = pairNum + 1

NextPair:
        Next j

        ' 最後の発言が単独（回答なし）の場合も残す
        lastIdx = UBound(parts)
        startIdx = IIf(isRenraku, 1, 0)
        adjustedLast = lastIdx - startIdx

        If adjustedLast Mod 2 = 0 And adjustedLast > 0 Then
            sp1 = Trim(parts(lastIdx))
            If isRenraku Then
                If lastIdx Mod 2 = 0 Then
                    role1 = roleOP
                Else
                    role1 = roleCL
                End If
            Else
                If lastIdx Mod 2 = 0 Then
                    role1 = roleCL
                Else
                    role1 = roleOP
                End If
            End If

            wsSplit.Cells(outRow, 1).Value = wsMask.Cells(i, 1).Value
            wsSplit.Cells(outRow, 2).Value = wsMask.Cells(i, 10).Value
            wsSplit.Cells(outRow, 3).Value = pairNum
            wsSplit.Cells(outRow, 4).Value = role1
            wsSplit.Cells(outRow, 5).Value = sp1
            wsSplit.Cells(outRow, 6).Value = ""
            wsSplit.Cells(outRow, 7).Value = ""
            wsSplit.Rows(outRow).Font.Name = "Arial"
            wsSplit.Rows(outRow).Font.Size = 9
            wsSplit.Cells(outRow, 5).WrapText = True
            wsSplit.Rows(outRow).RowHeight = 45
            outRow = outRow + 1
        End If

NextRow:
    Next i

    Application.ScreenUpdating = True

    MsgBox outRow - 3 & " ken no bunkatsu ga kanryo shimashita."
End Sub

' ----------------------------------------------------------------
' STEP4: AI送付用CSVエクスポート
' ----------------------------------------------------------------
Sub ExportCSV()
    Dim ws As Worksheet
    Dim fPath As Variant
    Dim fNum As Integer
    Dim lastRow As Long
    Dim i As Long
    Dim j As Integer
    Dim line As String
    Dim val As String

    Set ws = ThisWorkbook.Sheets("Split")

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox "Error: No data in Split sheet. Please run split first."
        Exit Sub
    End If

    fPath = Application.GetSaveAsFilename( _
        InitialFileName:="FAQ_" & Format(Now, "YYYYMMDD") & ".csv", _
        FileFilter:="CSV," & Chr(42) & ".csv")
    If VarType(fPath) = vbBoolean Then Exit Sub

    fNum = FreeFile
    Open fPath For Output As #fNum

    ' ヘッダー行
    Print #fNum, "ID,Category,PairNo,Role1,Text1,Role2,Text2"

    For i = 3 To lastRow
        line = ""
        For j = 1 To 7
            val = CStr(ws.Cells(i, j).Value)
            val = Replace(val, Chr(34), Chr(34) & Chr(34))
            If InStr(val, Chr(44)) > 0 Or InStr(val, Chr(10)) > 0 Or InStr(val, Chr(34)) > 0 Then
                val = Chr(34) & val & Chr(34)
            End If
            If j = 1 Then
                line = val
            Else
                line = line & Chr(44) & val
            End If
        Next j
        Print #fNum, line
    Next i

    Close #fNum

    MsgBox "CSV wo hozon shimashita."
End Sub
