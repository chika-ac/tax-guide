    On Error GoTo 0
    Dim fso As Object, ts As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.OpenTextFile(fPath, 1, False, -2)
    Dim ln As Long: ln = 1
    Do While Not ts.AtEndOfStream
        Dim lineStr As String
        lineStr = ts.ReadLine
        If Trim(lineStr) = "" Then GoTo NextLine   ' 空行スキップ
        Dim cols() As String
        cols = SplitCSV(lineStr)
        Dim i As Integer
        For i = 0 To UBound(cols)
            If Trim(cols(i)) <> "" Then
                ws.Cells(ln, i + 1).Value = Trim(cols(i))
            End If
        Next i
        ln = ln + 1
NextLine:
    Loop
    ts.Close
    r = ln
    GoTo Done
