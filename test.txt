function runLinkCheck() {
  const allApps = [...regularApps, ...eventApps, ...internalApps];
  const container = document.getElementById("checkResults");
  container.innerHTML = "";

  allApps.forEach(app => {
    const item = document.createElement("div");
    item.className = "check-item";
    item.innerHTML = `
      <span style="font-size:13px;flex:1;">${app.name}</span>
      <a href="${app.url}" target="_blank" style="font-size:12px;color:var(--accent);">開いて確認 →</a>
    `;
    container.appendChild(item);
  });

  const summary = document.createElement("p");
  summary.style.cssText = "font-size:13px;margin-top:1rem;color:var(--muted);";
  summary.textContent = "各リンクを開いて到達できるか確認してください。";
  container.appendChild(summary);
}


<button class="btn-check" id="runCheckBtn" onclick="runLinkCheck()">リンク一覧を表示</button>
