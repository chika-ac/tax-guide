Dim figuresSource As String
Dim figuresDev As String
Dim figuresProd As String
figuresSource = basePath & "\..\assets\figures"
figuresDev    = basePath & "\..\dev\assets\figures"
figuresProd   = basePath & "\..\prod\assets\figures"
If fso2.FolderExists(figuresSource) Then
    If Not fso2.FolderExists(figuresDev)  Then fso2.CreateFolder figuresDev
    If Not fso2.FolderExists(figuresProd) Then fso2.CreateFolder figuresProd
    Dim oFile2 As Object
    For Each oFile2 In fso2.GetFolder(figuresSource).Files
        fso2.CopyFile oFile2.Path, figuresDev  & "\" & oFile2.Name, True
        fso2.CopyFile oFile2.Path, figuresProd & "\" & oFile2.Name, True
    Next oFile2
End If
