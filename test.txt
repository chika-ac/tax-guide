Sub ExportDataJs()

' dev → prod ファイルコピー
    Dim fso3 As Object
    Dim devFolder As String
    Dim prodFolder As String
    Set fso3 = CreateObject("Scripting.FileSystemObject")
    devFolder  = basePath & "\..\dev"
    prodFolder = basePath & "\..\prod"
    If Not fso3.FolderExists(prodFolder) Then fso3.CreateFolder prodFolder
    fso3.CopyFile devFolder & "\app.js",     prodFolder & "\app.js",     True
    fso3.CopyFile devFolder & "\styles.css", prodFolder & "\styles.css", True
    fso3.CopyFile devFolder & "\index.html", prodFolder & "\index.html", True
    Set fso3 = Nothing

End Sub
