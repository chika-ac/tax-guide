' ----------------------------------------------------------------
' STEP1: CSVインポート
' ----------------------------------------------------------------
Sub ImportCSV()
    Dim ws As Worksheet
    Dim fPath As Variant
    Dim fNum As Integer
    Dim line As String
    Dim fields() As String
    Dim rowNum As Long
    Dim isFirst As Boolean
    Dim i As Integer
    Dim lastRow As Long
    Dim msg As String

    fPath = Application.GetOpenFilename("CSV," & Chr(42) & ".csv")
    If VarType(fPath) = vbBoolean Then Exit Sub

    Set ws = ThisWorkbook.Sheets(Chr(20803) & Chr(12487) & Chr(12540) & Chr(12479))

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow > 2 Then
        ws.Rows("3:" & lastRow).ClearContents
    End If

    fNum = FreeFile
    Open fPath For Input As #fNum
    rowNum = 3
    isFirst = True

    Do While Not EOF(fNum)
        Line Input #fNum, line
        If isFirst Then
            isFirst = False
        Else
            fields = ParseCSVLine(line)
            If UBound(fields) >= 0 Then
                If Trim(fields(0)) <> "" Then
                    For i = 0 To UBound(fields)
                        If i < 23 Then
                            ws.Cells(rowNum, i + 1).Value = fields(i)
                        End If
                    Next i
                    rowNum = rowNum + 1
                End If
            End If
        End If
    Loop

    Close #fNum

    msg = rowNum - 3 & " " & Chr(20855) & Chr(12398) & Chr(12487) & Chr(12540) & Chr(12479) & Chr(12434) & Chr(12452) & Chr(12531) & Chr(12509) & Chr(12540) & Chr(12488) & Chr(12375) & Chr(12414) & Chr(12375) & Chr(12383) & Chr(12290)
    MsgBox msg
End Sub

' ----------------------------------------------------------------
' CSVライン解析（ダブルクォート・カンマ対応）
' ----------------------------------------------------------------
Function ParseCSVLine(line As String) As String()
    Dim result() As String
    Dim count As Integer
    Dim i As Long
    Dim c As String
    Dim current As String
    Dim inQuote As Boolean

    count = 0
    ReDim result(0)
    current = ""
    inQuote = False

    For i = 1 To Len(line)
        c = Mid(line, i, 1)
        If c = Chr(34) Then
            If inQuote And i < Len(line) And Mid(line, i + 1, 1) = Chr(34) Then
                current = current & Chr(34)
                i = i + 1
            Else
                inQuote = Not inQuote
            End If
        ElseIf c = Chr(44) And Not inQuote Then
            result(count) = current
            count = count + 1
            ReDim Preserve result(count)
            current = ""
        Else
            current = current & c
        End If
    Next i
    result(count) = current
    ParseCSVLine = result
End Function

' ----------------------------------------------------------------
' STEP2: マスク実行
' ----------------------------------------------------------------
Sub RunMask()
    Dim wsSrc As Worksheet
    Dim wsMask As Worksheet
    Dim lastRow As Long
    Dim lastRow2 As Long
    Dim i As Long
    Dim j As Integer
    Dim cellVal As String
    Dim maskChar As String
    Dim msg2 As String

    Set wsSrc = ThisWorkbook.Sheets(Chr(20803) & Chr(12487) & Chr(12540) & Chr(12479))
    Set wsMask = ThisWorkbook.Sheets(Chr(12510) & Chr(12473) & Chr(12463) & Chr(28168) & Chr(12487) & Chr(12540) & Chr(12479))
    maskChar = Chr(9679) & Chr(9679)

    lastRow = wsSrc.Cells(wsSrc.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox Chr(20803) & Chr(12487) & Chr(12540) & Chr(12479) & Chr(12364) & Chr(35211) & Chr(35211) & Chr(12354) & Chr(12426) & Chr(12414) & Chr(12379) & Chr(12435) & Chr(12290)
        Exit Sub
    End If

    lastRow2 = wsMask.Cells(wsMask.Rows.Count, 1).End(xlUp).Row
    If lastRow2 > 2 Then
        wsMask.Rows("3:" & lastRow2).ClearContents
    End If

    Application.ScreenUpdating = False

    For i = 3 To lastRow
        For j = 1 To 23
            cellVal = CStr(wsSrc.Cells(i, j).Value)
            cellVal = MaskText(cellVal, maskChar)
            wsMask.Cells(i, j).Value = cellVal
            wsMask.Cells(i, j).Font.Name = "Arial"
            wsMask.Cells(i, j).Font.Size = 9
        Next j
    Next i

    Application.ScreenUpdating = True

    msg2 = lastRow - 2 & " " & Chr(20855) & Chr(12398) & Chr(12510) & Chr(12473) & Chr(12463) & Chr(20966) & Chr(29702) & Chr(12364) & Chr(23436) & Chr(20102) & Chr(12375) & Chr(12414) & Chr(12375) & Chr(12383) & Chr(12290)
    MsgBox msg2
End Sub

' ----------------------------------------------------------------
' テキストマスク処理（正規表現）
' ----------------------------------------------------------------
Function MaskText(text As String, maskChar As String) As String
    Dim re As Object
    Dim result As String
    result = text

    Set re = CreateObject("VBScript.RegExp")
    re.Global = True
    re.IgnoreCase = False

    ' 申請番号（12桁）
    re.Pattern = "\b\d{12}\b"
    result = re.Replace(result, maskChar)

    ' 職員番号（7〜8桁）
    re.Pattern = "\b\d{7,8}\b"
    result = re.Replace(result, maskChar)

    ' 氏名＋敬称
    re.Pattern = "[^\s\n]{1,10}(" & Chr(27096) & "|" & Chr(12373) & Chr(12414) & "|" & Chr(12469) & Chr(12510) & "|" & Chr(12373) & Chr(12435) & ")"
    result = re.Replace(result, maskChar)

    ' 年収（円・えん）
    re.Pattern = "\d[\d,]*(" & Chr(20870) & "|" & Chr(12360) & Chr(12435) & ")"
    result = re.Replace(result, maskChar)

    ' 年収（¥記号）
    re.Pattern = Chr(165) & "[\d,]+"
    result = re.Replace(result, maskChar)

    MaskText = result
End Function

' ----------------------------------------------------------------
' STEP3: 会話分割
' ----------------------------------------------------------------
Sub RunSplit()
    Dim wsMask As Worksheet
    Dim wsSplit As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim outRow As Long
    Dim j As Integer
    Dim memo As String
    Dim parts() As String
    Dim sp1 As String
    Dim sp2 As String
    Dim role1 As String
    Dim role2 As String
    Dim pairNum As Integer
    Dim isRenraku As Boolean
    Dim startIdx As Integer
    Dim lastIdx As Integer
    Dim adjustedLast As Integer
    Dim msg3 As String

    Set wsMask = ThisWorkbook.Sheets(Chr(12510) & Chr(12473) & Chr(12463) & Chr(28168) & Chr(12487) & Chr(12540) & Chr(12479))
    Set wsSplit = ThisWorkbook.Sheets(Chr(20998) & Chr(21106) & Chr(28168) & Chr(12487) & Chr(12540) & Chr(12479))

    lastRow = wsMask.Cells(wsMask.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox Chr(12510) & Chr(12473) & Chr(12463) & Chr(28168) & Chr(12487) & Chr(12540) & Chr(12479) & Chr(12364) & Chr(35211) & Chr(35211) & Chr(12354) & Chr(12426) & Chr(12414) & Chr(12379) & Chr(12435) & Chr(12290)
        Exit Sub
    End If

    Dim lastRow3 As Long
    lastRow3 = wsSplit.Cells(wsSplit.Rows.Count, 1).End(xlUp).Row
    If lastRow3 > 2 Then
        wsSplit.Rows("3:" & lastRow3).ClearContents
    End If

    Application.ScreenUpdating = False
    outRow = 3

    For i = 3 To lastRow
        memo = wsMask.Cells(i, 12).Value
        If Trim(memo) = "" Then GoTo NextRow

        ' 改行+↓+改行 で分割、なければ↓のみで分割
        If InStr(memo, Chr(10) & Chr(8595) & Chr(10)) > 0 Then
            parts = Split(memo, Chr(10) & Chr(8595) & Chr(10))
        Else
            parts = Split(memo, Chr(8595))
        End If

        ' 連携発信チェック
        isRenraku = (InStr(parts(0), Chr(36899) & Chr(25345) & Chr(30330) & Chr(20449)) > 0)

        pairNum = 1

        For j = 0 To UBound(parts) - 1
            If isRenraku And j = 0 Then GoTo NextPair

            sp1 = Trim(parts(j))
            sp2 = Trim(parts(j + 1))

            If isRenraku Then
                If j Mod 2 = 0 Then
                    role1 = Chr(12458) & Chr(12506) & Chr(12524) & Chr(12540) & Chr(12479) & Chr(12540)
                    role2 = Chr(12463) & Chr(12521) & Chr(12452) & Chr(12450) & Chr(12531) & Chr(12488)
                Else
                    role1 = Chr(12463) & Chr(12521) & Chr(12452) & Chr(12450) & Chr(12531) & Chr(12488)
                    role2 = Chr(12458) & Chr(12506) & Chr(12524) & Chr(12540) & Chr(12479) & Chr(12540)
                End If
            Else
                If j Mod 2 = 0 Then
                    role1 = Chr(12463) & Chr(12521) & Chr(12452) & Chr(12450) & Chr(12531) & Chr(12488)
                    role2 = Chr(12458) & Chr(12506) & Chr(12524) & Chr(12540) & Chr(12479) & Chr(12540)
                Else
                    role1 = Chr(12458) & Chr(12506) & Chr(12524) & Chr(12540) & Chr(12479) & Chr(12540)
                    role2 = Chr(12463) & Chr(12521) & Chr(12452) & Chr(12450) & Chr(12531) & Chr(12488)
                End If
            End If

            wsSplit.Cells(outRow, 1).Value = wsMask.Cells(i, 1).Value
            wsSplit.Cells(outRow, 2).Value = wsMask.Cells(i, 10).Value
            wsSplit.Cells(outRow, 3).Value = pairNum
            wsSplit.Cells(outRow, 4).Value = role1
            wsSplit.Cells(outRow, 5).Value = sp1
            wsSplit.Cells(outRow, 6).Value = role2
            wsSplit.Cells(outRow, 7).Value = sp2
            wsSplit.Rows(outRow).Font.Name = "Arial"
            wsSplit.Rows(outRow).Font.Size = 9
            wsSplit.Cells(outRow, 5).WrapText = True
            wsSplit.Cells(outRow, 7).WrapText = True
            wsSplit.Rows(outRow).RowHeight = 45

            outRow = outRow + 1
            pairNum = pairNum + 1

NextPair:
        Next j

        ' 最後の発言が単独（回答なし）の場合も残す
        lastIdx = UBound(parts)
        startIdx = IIf(isRenraku, 1, 0)
        adjustedLast = lastIdx - startIdx

        If adjustedLast Mod 2 = 0 And adjustedLast > 0 Then
            sp1 = Trim(parts(lastIdx))
            If isRenraku Then
                If lastIdx Mod 2 = 0 Then
                    role1 = Chr(12458) & Chr(12506) & Chr(12524) & Chr(12540) & Chr(12479) & Chr(12540)
                Else
                    role1 = Chr(12463) & Chr(12521) & Chr(12452) & Chr(12450) & Chr(12531) & Chr(12488)
                End If
            Else
                If lastIdx Mod 2 = 0 Then
                    role1 = Chr(12463) & Chr(12521) & Chr(12452) & Chr(12450) & Chr(12531) & Chr(12488)
                Else
                    role1 = Chr(12458) & Chr(12506) & Chr(12524) & Chr(12540) & Chr(12479) & Chr(12540)
                End If
            End If

            wsSplit.Cells(outRow, 1).Value = wsMask.Cells(i, 1).Value
            wsSplit.Cells(outRow, 2).Value = wsMask.Cells(i, 10).Value
            wsSplit.Cells(outRow, 3).Value = pairNum
            wsSplit.Cells(outRow, 4).Value = role1
            wsSplit.Cells(outRow, 5).Value = sp1
            wsSplit.Cells(outRow, 6).Value = ""
            wsSplit.Cells(outRow, 7).Value = ""
            wsSplit.Rows(outRow).Font.Name = "Arial"
            wsSplit.Rows(outRow).Font.Size = 9
            wsSplit.Cells(outRow, 5).WrapText = True
            wsSplit.Rows(outRow).RowHeight = 45
            outRow = outRow + 1
        End If

NextRow:
    Next i

    Application.ScreenUpdating = True

    msg3 = outRow - 3 & " " & Chr(20855) & Chr(12398) & Chr(30330) & Chr(35328) & Chr(12506) & Chr(12450) & Chr(12434) & Chr(29983) & Chr(25104) & Chr(12375) & Chr(12414) & Chr(12375) & Chr(12383) & Chr(12290)
    MsgBox msg3
End Sub

    Dim fPath As Variant
    Dim fNum As Integer
    Dim lastRow As Long
    Dim i As Long
    Dim j As Integer
    Dim line As String
    Dim val As String
    Dim msg4 As String

    Set ws = ThisWorkbook.Sheets(Chr(20998) & Chr(21106) & Chr(28168) & Chr(12487) & Chr(12540) & Chr(12479))

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox Chr(20998) & Chr(21106) & Chr(28168) & Chr(12487) & Chr(12540) & Chr(12479) & Chr(12364) & Chr(35211) & Chr(35211) & Chr(12354) & Chr(12426) & Chr(12414) & Chr(12379) & Chr(12435) & Chr(12290)
        Exit Sub
    End If

    fPath = Application.GetSaveAsFilename( _
        InitialFileName:="FAQ" & Chr(36865) & Chr(20184) & Chr(29992) & "_" & Format(Now, "YYYYMMDD") & ".csv", _
        FileFilter:="CSV," & Chr(42) & ".csv")
    If VarType(fPath) = vbBoolean Then Exit Sub

    fNum = FreeFile
    Open fPath For Output As #fNum

    Print #fNum, Chr(21839) & Chr(21512) & Chr(12379) & Chr(30058) & Chr(21495) & Chr(44) & Chr(26989) & Chr(21153) & Chr(20998) & Chr(39006) & Chr(35443) & Chr(32048) & Chr(44) & Chr(30330) & Chr(35328) & Chr(30058) & Chr(21495) & Chr(44) & Chr(30330) & Chr(35328) & Chr(32773) & Chr(9312) & Chr(44) & Chr(30330) & Chr(35328) & Chr(20869) & Chr(23481) & Chr(9312) & Chr(44) & Chr(30330) & Chr(35328) & Chr(32773) & Chr(9313) & Chr(44) & Chr(30330) & Chr(35328) & Chr(20869) & Chr(23481) & Chr(9313)

    For i = 3 To lastRow
        line = ""
        For j = 1 To 7
            val = CStr(ws.Cells(i, j).Value)
            val = Replace(val, Chr(34), Chr(34) & Chr(34))
            If InStr(val, Chr(44)) > 0 Or InStr(val, Chr(10)) > 0 Or InStr(val, Chr(34)) > 0 Then
                val = Chr(34) & val & Chr(34)
            End If
            If j = 1 Then
                line = val
            Else
                line = line & Chr(44) & val
            End If
        Next j
        Print #fNum, line
    Next i

    Close #fNum

    msg4 = "CSV" & Chr(12434) & Chr(20445) & Chr(23384) & Chr(12375) & Chr(12414) & Chr(12375) & Chr(12383) & Chr(12290)
    MsgBox msg4
End Sub
