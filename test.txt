
' =====================================================================
' コールセンター シミュレーション v3
' 着信総数 = 受電数 + 新規折り返し待ち を保証
' =====================================================================

Sub RunSimulation()
    Dim wsParam  As Worksheet: Set wsParam  = ThisWorkbook.Sheets("パラメータ")
    Dim wsInput  As Worksheet: Set wsInput  = ThisWorkbook.Sheets("入力データ")
    Dim wsResult As Worksheet: Set wsResult = ThisWorkbook.Sheets("結果")
    Dim wsDetail As Worksheet: Set wsDetail = ThisWorkbook.Sheets("シミュレーション明細")

    ' ---- パラメータ読み込み ----
    Dim numOp    As Long: numOp    = CLng(wsParam.Range("C5").Value)
    Dim postProc As Long: postProc = CLng(wsParam.Range("C6").Value)
    Dim cbTalk   As Long: cbTalk   = CLng(wsParam.Range("C7").Value)
    Dim rkPrep   As Long: rkPrep   = CLng(wsParam.Range("C8").Value)
    Dim rkTalk   As Long: rkTalk   = CLng(wsParam.Range("C9").Value)
    Dim rkTotal  As Long: rkTotal  = rkPrep + rkTalk
    Dim dayStart As Long: dayStart = HMtoMin(CStr(wsParam.Range("C10").Value))
    Dim dayEnd   As Long: dayEnd   = HMtoMin(CStr(wsParam.Range("C11").Value))
    Dim dtStart  As Date: dtStart  = CDate(wsParam.Range("C12").Value)
    Dim dtEnd    As Date: dtEnd    = CDate(wsParam.Range("C13").Value)

    ' ---- 日別連携発信件数 ----
    Dim rkDates(1 To 100) As Date
    Dim rkCnts(1 To 100)  As Long
    Dim rkN As Long: rkN = 0
    Dim r As Long
    For r = 17 To 47
        Dim dv As Variant: dv = wsParam.Cells(r, 2).Value
        Dim cv As Variant: cv = wsParam.Cells(r, 3).Value
        If dv <> "" And cv <> "" Then
            Dim td As Date
            On Error Resume Next
            td = CDate(dv)
            On Error GoTo 0
            If td > 0 Then
                rkN = rkN + 1
                rkDates(rkN) = td
                rkCnts(rkN) = CLng(cv)
            End If
        End If
    Next r

    ' ---- 入力データ読み込み ----
    Dim lastRow As Long
    lastRow = wsInput.Cells(wsInput.Rows.Count, 2).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox "入力データがありません。", vbExclamation: Exit Sub
    End If

    Const INT_NUM As String = "4140333"

    ReDim evDate(1 To 10000) As Date
    ReDim evMin(1 To 10000)  As Long
    ReDim evDur(1 To 10000)  As Long
    Dim evN As Long: evN = 0

    For r = 3 To lastRow
        Dim sType As String: sType = CQ(CStr(wsInput.Cells(r, 2).Value))
        If sType <> "着信" Then GoTo Skip

        Dim sF As String: sF = L7(CQ(CStr(wsInput.Cells(r, 3).Value)))
        Dim sT As String: sT = L7(CQ(CStr(wsInput.Cells(r, 4).Value)))
        If sF = INT_NUM And sT = INT_NUM Then GoTo Skip

        Dim sD As String: sD = CQ(CStr(wsInput.Cells(r, 5).Value))
        If sD = "" Then GoTo Skip
        Dim dtR As Date
        On Error Resume Next
        dtR = CDate(sD)
        On Error GoTo 0
        If dtR = 0 Then GoTo Skip
        If DateValue(dtR) < dtStart Or DateValue(dtR) > dtEnd Then GoTo Skip

        Dim sB As String: sB = CQ(CStr(wsInput.Cells(r, 7).Value))
        Dim sE As String: sE = CQ(CStr(wsInput.Cells(r, 8).Value))
        Dim dur As Long: dur = 0
        If sB <> "" And sE <> "" Then
            On Error Resume Next
            dur = CLng((CDate(sE) - CDate(sB)) * 24 * 60)
            On Error GoTo 0
            If dur < 0 Then dur = 0
        End If

        evN = evN + 1
        evDate(evN) = DateValue(dtR)
        evMin(evN)  = Hour(dtR) * 60 + Minute(dtR)
        evDur(evN)  = dur
Skip:
    Next r

    ' ---- 着信データを日付・時刻順にソート（バブルソート）----
    Dim si As Long, sj As Long
    Dim tmpD As Date, tmpM As Long, tmpDur As Long
    For si = 1 To evN - 1
        For sj = 1 To evN - si
            Dim a As Long: a = sj
            Dim b As Long: b = sj + 1
            Dim aKey As Double: aKey = CDbl(evDate(a)) * 1440 + evMin(a)
            Dim bKey As Double: bKey = CDbl(evDate(b)) * 1440 + evMin(b)
            If aKey > bKey Then
                tmpD = evDate(a): evDate(a) = evDate(b): evDate(b) = tmpD
                tmpM = evMin(a):  evMin(a)  = evMin(b):  evMin(b)  = tmpM
                tmpDur = evDur(a): evDur(a) = evDur(b): evDur(b) = tmpDur
            End If
        Next sj
    Next si

    ' ---- シートクリア ----
    wsResult.Rows("4:" & wsResult.Rows.Count).ClearContents
    wsResult.Rows("4:" & wsResult.Rows.Count).Interior.ColorIndex = xlNone
    wsDetail.Rows("4:" & wsDetail.Rows.Count).ClearContents
    wsDetail.Rows("4:" & wsDetail.Rows.Count).Interior.ColorIndex = xlNone

    Dim resRow As Long: resRow = 4
    Dim detRow As Long: detRow = 4
    Dim cbQueue As Long: cbQueue = 0
    Dim rkQueue As Long: rkQueue = 0

    ' ---- 日ごとシミュレーション ----
    Dim dt As Date
    For dt = dtStart To dtEnd

        ' この日の着信を収集（ソート済みなので順序保証）
        Dim dEvN As Long: dEvN = 0
        ReDim dMin(1 To 5000) As Long
        ReDim dDur(1 To 5000) As Long
        Dim i As Long
        For i = 1 To evN
            If evDate(i) = dt Then
                dEvN = dEvN + 1
                dMin(dEvN) = evMin(i)
                dDur(dEvN) = evDur(i)
            End If
        Next i

        ' この日の連携発信指定件数
        Dim rkDay As Long: rkDay = 0
        For i = 1 To rkN
            If rkDates(i) = dt Then rkDay = rkCnts(i): Exit For
        Next i

        ' オペレーター解放時刻
        ReDim opFree(1 To numOp) As Long
        For i = 1 To numOp: opFree(i) = dayStart: Next i

        Dim inTotal As Long: inTotal = dEvN
        Dim inAns   As Long: inAns   = 0
        Dim newCB   As Long: newCB   = 0
        Dim cbDone  As Long: cbDone  = 0
        Dim rkDone  As Long: rkDone  = 0

        Dim pendCB As Long: pendCB = cbQueue: cbQueue = 0
        Dim pendRK As Long: pendRK = rkQueue + rkDay: rkQueue = 0

        Dim evPtr As Long: evPtr = 1
        Dim guard As Long: guard = 0

        ' ---- イベントループ ----
        Do
            guard = guard + 1
            If guard > 50000 Then Exit Do

            ' 次の着信時刻
            Dim nextIn As Long
            If evPtr <= dEvN Then nextIn = dMin(evPtr) Else nextIn = dayEnd + 1

            ' 最も早く空くオペレーターと時刻
            Dim bOp As Long: bOp = 1
            Dim bMin As Long: bMin = opFree(1)
            For i = 2 To numOp
                If opFree(i) < bMin Then bMin = opFree(i): bOp = i
            Next i

            ' ========================================
            ' 着信が来る前にオペレーターが空いている
            ' → 隙間時間：折り返し or 連携発信
            ' ========================================
            If bMin < nextIn And bMin < dayEnd Then

                If pendCB > 0 Then
                    ' 折り返し発信が隙間に収まるか確認
                    Dim cbEnd As Long: cbEnd = bMin + cbTalk
                    ' 収まらなくても次の着信前までに終わるなら実施
                    If cbEnd <= nextIn Or nextIn > dayEnd Then
                        If cbEnd <= dayEnd Then
                            pendCB = pendCB - 1: cbDone = cbDone + 1
                            opFree(bOp) = cbEnd
                            WD wsDetail, detRow, dt, "折り返し発信", MTS(bMin), MTS(cbEnd), "", cbTalk, "", "折り返しキューから処理"
                            detRow = detRow + 1
                        Else
                            cbQueue = cbQueue + pendCB: pendCB = 0
                        End If
                    Else
                        ' 次の着信までに終わらない → 着信処理を優先
                        GoTo HandleInbound
                    End If

                ElseIf pendRK > 0 Then
                    Dim rkEnd As Long: rkEnd = bMin + rkTotal
                    If rkEnd <= nextIn Or nextIn > dayEnd Then
                        If rkEnd <= dayEnd Then
                            pendRK = pendRK - 1: rkDone = rkDone + 1
                            opFree(bOp) = rkEnd
                            WD wsDetail, detRow, dt, "連携発信", MTS(bMin), MTS(bMin + rkPrep), "", rkTalk, MTS(rkEnd), "準備" & rkPrep & "分+通話" & rkTalk & "分"
                            detRow = detRow + 1
                        Else
                            rkQueue = rkQueue + pendRK: pendRK = 0
                        End If
                    Else
                        GoTo HandleInbound
                    End If

                Else
                    ' 隙間だがやることなし → 着信待ちへ
                    If evPtr > dEvN Then Exit Do
                    GoTo HandleInbound
                End If

            ' ========================================
            ' 着信処理
            ' ========================================
            Else
HandleInbound:
                If evPtr > dEvN Then Exit Do

                Dim arrMin As Long: arrMin = dMin(evPtr)
                Dim durM   As Long: durM   = dDur(evPtr)
                evPtr = evPtr + 1

                ' 着信到着時点で空きオペレーターを探す
                Dim found As Long: found = 0
                Dim fMin2 As Long: fMin2 = 99999
                For i = 1 To numOp
                    If opFree(i) <= arrMin Then
                        If opFree(i) < fMin2 Then fMin2 = opFree(i): found = i
                    End If
                Next i

                If found = 0 Then
                    ' 全員塞がり → 必ず折り返し待ちへ
                    pendCB = pendCB + 1: newCB = newCB + 1
                    WD wsDetail, detRow, dt, "着信→折り返し待ち", MTS(arrMin), "", "", 0, "", "全員対応中"
                    detRow = detRow + 1
                Else
                    ' 受電 → 通話 → 後続処理
                    Dim tEnd As Long: tEnd = arrMin + durM
                    Dim pEnd As Long: pEnd = tEnd + postProc
                    If pEnd > dayEnd Then pEnd = dayEnd
                    opFree(found) = pEnd
                    inAns = inAns + 1
                    WD wsDetail, detRow, dt, "着信受電", MTS(arrMin), MTS(tEnd), MTS(arrMin), durM, MTS(pEnd), "受電→後続処理完了"
                    detRow = detRow + 1
                End If
            End If

            If detRow > 50000 Then Exit Do
        Loop

        ' 未処理を翌日繰越
        cbQueue = cbQueue + pendCB
        rkQueue = rkQueue + pendRK

        ' ---- 結果書き込み ----
        With wsResult
            .Cells(resRow, 2).Value = dt: .Cells(resRow, 2).NumberFormat = "yyyy/mm/dd"
            .Cells(resRow, 3).Value = inTotal   ' 着信総数
            .Cells(resRow, 4).Value = inAns     ' 受電数
            .Cells(resRow, 5).Value = newCB     ' 新規折り返し待ち
            .Cells(resRow, 6).Value = cbDone    ' 折り返し発信完了
            .Cells(resRow, 7).Value = cbQueue   ' 折り返し発信残（繰越）
            .Cells(resRow, 8).Value = rkDone    ' 連携発信完了
            .Cells(resRow, 9).Value = rkQueue   ' 連携発信残（繰越）

            Dim busy As Double: busy = 0
            For i = 1 To numOp
                If opFree(i) > dayStart Then busy = busy + (opFree(i) - dayStart)
            Next i
            Dim cap As Double: cap = CDbl(numOp) * CDbl(dayEnd - dayStart)
            If cap > 0 Then .Cells(resRow, 10).Value = Round(busy / cap * 100, 1)
            .Cells(resRow, 10).NumberFormat = "0.0"

            If resRow Mod 2 = 0 Then
                For i = 2 To 10: .Cells(resRow, i).Interior.Color = RGB(235, 243, 255): Next i
            End If
        End With

        resRow = resRow + 1
    Next dt

    ' 合計行
    If resRow > 4 Then
        With wsResult
            .Cells(resRow, 2).Value = "合計"
            .Cells(resRow, 2).Font.Bold = True
            .Cells(resRow, 2).Font.Name = "Arial"
            Dim sc As Variant
            For Each sc In Array(3, 4, 5, 6, 7, 8, 9)
                .Cells(resRow, sc).Formula = "=SUM(" & .Cells(4, sc).Address & ":" & .Cells(resRow - 1, sc).Address & ")"
                .Cells(resRow, sc).Font.Bold = True
            Next sc
            For i = 2 To 10: .Cells(resRow, i).Interior.Color = RGB(197, 217, 241): Next i
        End With
    End If

    MsgBox "シミュレーション完了！" & Chr(13) & Chr(13) & _
           "読み込み着信件数: " & evN & " 件" & Chr(13) & _
           "「結果」「シミュレーション明細」シートを確認してください。", _
           vbInformation, "完了"
End Sub

' =====================================================================
' ユーティリティ
' =====================================================================

Function CQ(s As String) As String
    s = Replace(s, "'", "")
    s = Trim(s)
    If s = Chr(34) Then s = ""
    CQ = s
End Function

Function L7(s As String) As String
    Dim cl As String: cl = ""
    Dim c As Long
    For c = 1 To Len(s)
        Dim ch As String: ch = Mid(s, c, 1)
        If ch >= "0" And ch <= "9" Then cl = cl & ch
    Next c
    If Len(cl) >= 7 Then L7 = Right(cl, 7) Else L7 = cl
End Function

Function HMtoMin(hhmm As String) As Long
    On Error Resume Next
    Dim p() As String: p = Split(hhmm, ":")
    If UBound(p) >= 1 Then HMtoMin = CLng(p(0)) * 60 + CLng(p(1)) Else HMtoMin = 510
    On Error GoTo 0
End Function

Function MTS(m As Long) As String
    If m <= 0 Then MTS = "" Else MTS = Format(m \ 60, "00") & ":" & Format(m Mod 60, "00")
End Function

Sub WD(ws As Worksheet, r As Long, dt As Date, evType As String, _
       stTime As String, edTime As String, origTime As String, _
       dur As Long, postTime As String, memo As String)
    With ws
        .Cells(r, 2).Value = dt: .Cells(r, 2).NumberFormat = "yyyy/mm/dd"
        .Cells(r, 3).Value = evType
        .Cells(r, 4).Value = stTime
        .Cells(r, 5).Value = edTime
        .Cells(r, 6).Value = origTime
        .Cells(r, 7).Value = dur
        .Cells(r, 8).Value = postTime
        .Cells(r, 9).Value = memo
        Dim i As Long
        For i = 2 To 9
            .Cells(r, i).Font.Name = "Arial"
            .Cells(r, i).Font.Size = 9
        Next i
        If r Mod 2 = 0 Then
            For i = 2 To 9: .Cells(r, i).Interior.Color = RGB(242, 242, 242): Next i
        End If
    End With
End Sub
