Sub FormatFAQ()
    Dim wsInput As Worksheet
    Dim wsFAQ As Worksheet
    Dim inputText As String
    Dim lines() As String
    Dim i As Long
    Dim outRow As Long
    Dim fields() As String
    Dim line As String

    Set wsInput = ThisWorkbook.Sheets("Input")
    Set wsFAQ = ThisWorkbook.Sheets("FAQ")

    ' InputシートのA1から全テキスト取得
    inputText = wsInput.Cells(1, 1).Value
    If Trim(inputText) = "" Then
        MsgBox "No data in Input sheet."
        Exit Sub
    End If

    ' FAQシートの既存データクリア（ヘッダー残す）
    Dim lastRowFAQ As Long
    lastRowFAQ = wsFAQ.Cells(wsFAQ.Rows.Count, 1).End(xlUp).Row
    If lastRowFAQ > 1 Then
        wsFAQ.Rows("2:" & lastRowFAQ).ClearContents
    End If

    ' 改行で分割
    If InStr(inputText, Chr(13) & Chr(10)) > 0 Then
        lines = Split(inputText, Chr(13) & Chr(10))
    Else
        lines = Split(inputText, Chr(10))
    End If

    Application.ScreenUpdating = False
    outRow = 2

    For i = 0 To UBound(lines)
        line = Trim(lines(i))
        If line = "" Then GoTo NextLine

        ' ヘッダー行はスキップ
        If line = Chr(34) & "分類" & Chr(34) & Chr(44) & Chr(34) & "Q" & Chr(34) & Chr(44) & Chr(34) & "A" & Chr(34) & Chr(44) & Chr(34) & "キーワード" & Chr(34) Then GoTo NextLine

        ' CSV解析
        fields = ParseCSVLine(line)

        If UBound(fields) >= 3 Then
            wsFAQ.Cells(outRow, 1).Value = fields(0)
            wsFAQ.Cells(outRow, 2).Value = fields(1)
            wsFAQ.Cells(outRow, 3).Value = fields(2)
            wsFAQ.Cells(outRow, 4).Value = fields(3)

            ' 書式
            wsFAQ.Rows(outRow).Font.Name = "Arial"
            wsFAQ.Rows(outRow).Font.Size = 9
            wsFAQ.Cells(outRow, 3).WrapText = True
            wsFAQ.Rows(outRow).RowHeight = 60

            outRow = outRow + 1
        End If

NextLine:
    Next i

    ' ヘッダー書式
    wsFAQ.Rows(1).Font.Name = "Arial"
    wsFAQ.Rows(1).Font.Size = 9
    wsFAQ.Rows(1).Font.Bold = True

    ' 列幅調整
    wsFAQ.Columns(1).ColumnWidth = 18
    wsFAQ.Columns(2).ColumnWidth = 35
    wsFAQ.Columns(3).ColumnWidth = 60
    wsFAQ.Columns(4).ColumnWidth = 25

    Application.ScreenUpdating = True

    MsgBox outRow - 2 & " ken no FAQ wo seikei shimashita."
End Sub
