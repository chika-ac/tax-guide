Sub RunSplit()
    Dim wsMask As Worksheet
    Dim wsSplit As Worksheet
    Dim lastRow As Long
    Dim lastRow3 As Long
    Dim i As Long
    Dim outRow As Long
    Dim memo As String
    Dim roleOP As String
    Dim roleCL As String

    Set wsMask = ThisWorkbook.Sheets("Masked")
    Set wsSplit = ThisWorkbook.Sheets("Split")

    roleOP = "OP"
    roleCL = "CL"

    lastRow = wsMask.Cells(wsMask.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        MsgBox "Error: No data in Masked sheet. Please run mask first."
        Exit Sub
    End If

    lastRow3 = wsSplit.Cells(wsSplit.Rows.Count, 1).End(xlUp).Row
    If lastRow3 > 2 Then
        wsSplit.Rows("3:" & lastRow3).ClearContents
    End If

    Application.ScreenUpdating = False
    outRow = 3

    Dim ARROW As String
    Dim HASSHIN As String
    Dim NYUGEN As String
    Dim RENRAKU As String
    ARROW = ChrW(8595)
    HASSHIN = ChrW(9632) & ChrW(30330) & ChrW(20449)
    NYUGEN = ChrW(9679) & ChrW(20837) & ChrW(38651)
    RENRAKU = ChrW(65308) & ChrW(36899) & ChrW(25345) & ChrW(30330) & ChrW(20449) & ChrW(65310)

    For i = 3 To lastRow
        memo = wsMask.Cells(i, 12).Value
        If Trim(memo) = "" Then GoTo NextRow

        ' ブロックに分割する
        ' 区切り文字：ARROW、HASSHIN、NYUGEN
        ' 各ブロックに発言者と内容を格納する

        Dim blocks() As String
        Dim roles() As String
        ReDim blocks(0)
        ReDim roles(0)
        Dim blockCount As Integer
        blockCount = 0

        Dim currentText As String
        Dim currentRole As String
        Dim pos As Long
        Dim nextPos As Long
        Dim sepLen As Integer
        Dim sepType As String
        Dim remaining As String

        remaining = memo
        currentText = ""
        currentRole = ""

        Do While Len(remaining) > 0
            ' 次の区切り文字を探す
            Dim posArrow As Long
            Dim posHasshin As Long
            Dim posNyugen As Long
            posArrow = InStr(remaining, ARROW)
            posHasshin = InStr(remaining, HASSHIN)
            posNyugen = InStr(remaining, NYUGEN)

            ' 最も近い区切り文字を選ぶ
            Dim minPos As Long
            minPos = 0

            If posArrow > 0 Then
                If minPos = 0 Or posArrow < minPos Then
                    minPos = posArrow
                    sepType = "ARROW"
                    sepLen = Len(ARROW)
                End If
            End If
            If posHasshin > 0 Then
                If minPos = 0 Or posHasshin < minPos Then
                    minPos = posHasshin
                    sepType = "HASSHIN"
                    sepLen = Len(HASSHIN)
                End If
            End If
            If posNyugen > 0 Then
                If minPos = 0 Or posNyugen < minPos Then
                    minPos = posNyugen
                    sepType = "NYUGEN"
                    sepLen = Len(NYUGEN)
                End If
            End If

            If minPos = 0 Then
                ' 区切り文字なし、残り全部が1ブロック
                currentText = currentText & remaining
                remaining = ""
            Else
                ' 区切り文字の手前までがcurrentText
                currentText = currentText & Left(remaining, minPos - 1)
                remaining = Mid(remaining, minPos + sepLen)

                ' currentTextをブロックとして保存
                If Trim(currentText) <> "" Then
                    ReDim Preserve blocks(blockCount)
                    ReDim Preserve roles(blockCount)
                    blocks(blockCount) = Trim(currentText)
                    roles(blockCount) = currentRole
                    blockCount = blockCount + 1
                End If

                ' 次のブロックの発言者を設定
                Select Case sepType
                    Case "HASSHIN"
                        currentRole = roleOP
                    Case "NYUGEN"
                        currentRole = roleCL
                    Case "ARROW"
                        ' 直前の発言者の逆
                        If currentRole = roleOP Then
                            currentRole = roleCL
                        ElseIf currentRole = roleCL Then
                            currentRole = roleOP
                        Else
                            ' 最初の発言者が不明の場合はCLと推定
                            currentRole = roleCL
                        End If
                End Select

                currentText = ""
            End If
        Loop

        ' 最後のブロックを保存
        If Trim(currentText) <> "" Then
            ReDim Preserve blocks(blockCount)
            ReDim Preserve roles(blockCount)
            blocks(blockCount) = Trim(currentText)
            roles(blockCount) = currentRole
            blockCount = blockCount + 1
        End If

        ' ブロックが1つ以下はスキップ
        If blockCount < 2 Then GoTo NextRow

        ' ペアを生成して出力
        Dim j As Integer
        Dim pairNum As Integer
        pairNum = 1

        For j = 0 To blockCount - 2
            Dim text1 As String
            Dim text2 As String
            Dim role1 As String
            Dim role2 As String
            text1 = blocks(j)
            text2 = blocks(j + 1)
            role1 = roles(j)
            role2 = roles(j + 1)

            ' 連携発信を含むペアは除外
            If InStr(text1, RENRAKU) > 0 Then GoTo NextPair
            If InStr(text2, RENRAKU) > 0 Then GoTo NextPair

            wsSplit.Cells(outRow, 1).Value = wsMask.Cells(i, 1).Value
            wsSplit.Cells(outRow, 2).Value = wsMask.Cells(i, 10).Value
            wsSplit.Cells(outRow, 3).Value = pairNum
            wsSplit.Cells(outRow, 4).Value = role1
            wsSplit.Cells(outRow, 5).Value = text1
            wsSplit.Cells(outRow, 6).Value = role2
            wsSplit.Cells(outRow, 7).Value = text2
            wsSplit.Rows(outRow).Font.Name = "Arial"
            wsSplit.Rows(outRow).Font.Size = 9
            wsSplit.Cells(outRow, 5).WrapText = True
            wsSplit.Cells(outRow, 7).WrapText = True
            wsSplit.Rows(outRow).RowHeight = 45

            outRow = outRow + 1
            pairNum = pairNum + 1

NextPair:
        Next j

        ' 最後のブロックが単独（回答なし）の場合も残す
        Dim lastBlock As String
        Dim lastRole As String
        lastBlock = blocks(blockCount - 1)
        lastRole = roles(blockCount - 1)

        ' 最後のブロックが連携発信を含む場合はスキップ
        If InStr(lastBlock, RENRAKU) > 0 Then GoTo NextRow

        ' blockCountが奇数の場合、最後のブロックは単独
        If blockCount Mod 2 = 1 Then
            wsSplit.Cells(outRow, 1).Value = wsMask.Cells(i, 1).Value
            wsSplit.Cells(outRow, 2).Value = wsMask.Cells(i, 10).Value
            wsSplit.Cells(outRow, 3).Value = pairNum
            wsSplit.Cells(outRow, 4).Value = lastRole
            wsSplit.Cells(outRow, 5).Value = lastBlock
            wsSplit.Cells(outRow, 6).Value = ""
            wsSplit.Cells(outRow, 7).Value = ""
            wsSplit.Rows(outRow).Font.Name = "Arial"
            wsSplit.Rows(outRow).Font.Size = 9
            wsSplit.Cells(outRow, 5).WrapText = True
            wsSplit.Rows(outRow).RowHeight = 45
            outRow = outRow + 1
        End If

NextRow:
    Next i

    Application.ScreenUpdating = True

    MsgBox outRow - 3 & " ken no bunkatsu ga kanryo shimashita."
End Sub
