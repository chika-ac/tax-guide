Sub 集計実行()
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    Dim t0 As Double
    t0 = Timer

    Call Build_Event_Table
    Call Build_Callback_Analysis
    Call Build_5Min_Callback
    Call Build_Aggregate
    Call Build_Summary

    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True

    MsgBox "集計完了しました（" & Format(Timer - t0, "0.0") & "秒）", vbInformation, "電話ログ分析ツール"
End Sub

' イベントテーブルから期間を取得
Function GetDateRange(ByRef dtStart As Date, ByRef dtEnd As Date) As Boolean
    Dim wsEV As Worksheet
    Set wsEV = ThisWorkbook.Sheets("イベントテーブル")
    If Not IsDate(wsEV.Cells(3, 2).Value) Or Not IsDate(wsEV.Cells(4, 2).Value) Then
        MsgBox "イベントテーブルのB3・B4に有効な日付を入力してください", vbExclamation
        GetDateRange = False
        Exit Function
    End If
    dtStart = CDate(wsEV.Cells(3, 2).Value)
    dtEnd   = CDate(wsEV.Cells(4, 2).Value)
    If dtStart > dtEnd Then
        MsgBox "開始日が終了日より後です", vbExclamation
        GetDateRange = False
        Exit Function
    End If
    GetDateRange = True
End Function

Function CleanStr(v As Variant) As String
    Dim s As String
    s = Trim(CStr(v))
    If Left(s, 1) = "'" Then s = Mid(s, 2)
    If Right(s, 1) = "'" Then s = Left(s, Len(s) - 1)
    CleanStr = Trim(s)
End Function

Function NormPhone(phone As String, digits As Integer) As String
    Dim p As String
    Dim i As Integer
    Dim ch As String
    p = ""
    For i = 1 To Len(phone)
        ch = Mid(phone, i, 1)
        If ch >= "0" And ch <= "9" Then p = p & ch
    Next i
    If Len(p) >= digits Then
        NormPhone = Right(p, digits)
    Else
        NormPhone = p
    End If
End Function

Function ParseDT(s As String) As Date
    On Error GoTo ErrH
    s = Trim(s)
    If s = "" Or s = Chr(34) Or s = "-" Then
        ParseDT = 0
        Exit Function
    End If
    ' yyyy/mm/dd h:mm:ss 形式を明示的に処理
    Dim spPos As Long
    spPos = InStr(s, " ")
    If spPos > 0 Then
        Dim datePart As String, timePart As String
        datePart = Left(s, spPos - 1)
        timePart = Mid(s, spPos + 1)
        Dim dParts() As String, tParts() As String
        dParts = Split(datePart, "/")
        tParts = Split(timePart, ":")
        If UBound(dParts) = 2 And UBound(tParts) >= 1 Then
            Dim yr As Long, mo As Long, dy As Long
            Dim hh As Long, mi As Long, ss As Long
            yr = CLng(dParts(0)): mo = CLng(dParts(1)): dy = CLng(dParts(2))
            hh = CLng(tParts(0)): mi = CLng(tParts(1))
            ss = 0
            If UBound(tParts) >= 2 Then ss = CLng(tParts(2))
            ParseDT = DateSerial(yr, mo, dy) + TimeSerial(hh, mi, ss)
            Exit Function
        End If
    End If
    ParseDT = CDate(s)
    Exit Function
ErrH:
    ParseDT = 0
End Function

Sub ReadInputData(ByRef data() As String, ByRef nRows As Long)
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("入力データ")
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 3 Then
        nRows = 0
        Exit Sub
    End If
    nRows = lastRow - 2
    ReDim data(1 To nRows, 1 To 8)
    Dim r As Long, c As Long
    For r = 1 To nRows
        For c = 1 To 8
            data(r, c) = CleanStr(ws.Cells(r + 2, c).Value)
        Next c
    Next r
End Sub

' ============================================================
' 処理①：イベントテーブル
' ============================================================
Sub Build_Event_Table()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("イベントテーブル")

    Dim dtStart As Date, dtEnd As Date
    If Not GetDateRange(dtStart, dtEnd) Then Exit Sub

    Dim data() As String, nRows As Long
    Call ReadInputData(data, nRows)
    If nRows = 0 Then
        MsgBox "入力データがありません", vbExclamation
        Exit Sub
    End If

    ' 5分スロット作成
    Dim tS As Date, tE As Date
    tS = TimeValue(TIME_START)
    tE = TimeValue(TIME_END)
    Dim nSlots As Long
    nSlots = 0
    Dim tt As Date
    tt = tS
    Do While tt <= tE
        nSlots = nSlots + 1
        tt = tt + TimeSerial(0, 5, 0)
    Loop
    ReDim slotTimes(1 To nSlots) As Date
    tt = tS
    Dim si As Long
    For si = 1 To nSlots
        slotTimes(si) = tt
        tt = tt + TimeSerial(0, 5, 0)
    Next si

    ' 日付リスト作成
    Dim nDays As Long
    nDays = DateDiff("d", dtStart, dtEnd) + 1
    ReDim dayList(1 To nDays) As Date
    Dim di As Long
    For di = 1 To nDays
        dayList(di) = dtStart + di - 1
    Next di

    ReDim bucketMax(1 To nSlots, 1 To nDays) As Long

    ' イベント収集
    Dim maxEv As Long
    maxEv = nRows * 2
    ReDim evTime(1 To maxEv) As Date
    ReDim evChange(1 To maxEv) As Long
    Dim nEv As Long
    nEv = 0
    Dim i As Long
    For i = 1 To nRows
        If data(i, 1) = "着信" Then
            Dim recvDT As Date, endDT As Date
            recvDT = ParseDT(data(i, 4))
            endDT  = ParseDT(data(i, 7))
            ' 期間外スキップ
            If recvDT > 0 Then
                Dim recvDate0 As Date: recvDate0 = Int(recvDT)
                If recvDate0 >= dtStart And recvDate0 <= dtEnd Then
                    nEv = nEv + 1
                    evTime(nEv) = recvDT
                    evChange(nEv) = 1
                    If endDT > 0 Then
                        nEv = nEv + 1
                        evTime(nEv) = endDT
                        evChange(nEv) = -1
                    End If
                End If
            End If
        End If
    Next i

    ' ソート
    Dim j As Long, tmpT As Date, tmpC As Long
    For i = 1 To nEv - 1
        For j = 1 To nEv - i
            Dim doSwap As Boolean
            doSwap = False
            If evTime(j) > evTime(j + 1) Then
                doSwap = True
            ElseIf evTime(j) = evTime(j + 1) And evChange(j) > evChange(j + 1) Then
                doSwap = True
            End If
            If doSwap Then
                tmpT = evTime(j): evTime(j) = evTime(j + 1): evTime(j + 1) = tmpT
                tmpC = evChange(j): evChange(j) = evChange(j + 1): evChange(j + 1) = tmpC
            End If
        Next j
    Next i

    ' 累積同時通話数
    Dim concurrent As Long
    concurrent = 0
    Dim ei As Long
    For ei = 1 To nEv
        concurrent = concurrent + evChange(ei)
        Dim evDate As Date
        evDate = Int(evTime(ei))
        Dim evTimeOnly As Date
        evTimeOnly = evTime(ei) - Int(evTime(ei))
        Dim evMin As Long
        evMin = Hour(evTimeOnly) * 60 + Minute(evTimeOnly)
        Dim slotMin As Long
        slotMin = (evMin \ 5) * 5
        Dim slotT As Date
        slotT = TimeSerial(slotMin \ 60, slotMin Mod 60, 0)
        Dim fsi As Long
        fsi = 0
        For si = 1 To nSlots
            If Abs(slotTimes(si) - slotT) < 0.00001 Then
                fsi = si: Exit For
            End If
        Next si
        Dim fdi As Long
        fdi = 0
        For di = 1 To nDays
            If dayList(di) = evDate Then
                fdi = di: Exit For
            End If
        Next di
        If fsi > 0 And fdi > 0 Then
            If concurrent > bucketMax(fsi, fdi) Then
                bucketMax(fsi, fdi) = concurrent
            End If
        End If
    Next ei

    ' シート出力
    ws.Rows("5:" & ws.Rows.Count).ClearContents
    ws.Rows("5:" & ws.Rows.Count).ClearFormats

    Dim maxAll As Long
    maxAll = 0
    For si = 1 To nSlots
        For di = 1 To nDays
            If bucketMax(si, di) > maxAll Then maxAll = bucketMax(si, di)
        Next di
    Next si
    Dim warnTh As Long, critTh As Long
    warnTh = CLng(maxAll * WARN_RATIO)
    critTh = CLng(maxAll * CRIT_RATIO)

    ' 行5ヘッダー
    Dim wdArr(0 To 6) As String
    wdArr(0) = "日": wdArr(1) = "月": wdArr(2) = "火"
    wdArr(3) = "水": wdArr(4) = "木": wdArr(5) = "金": wdArr(6) = "土"
    ws.Cells(5, 1).Value = "時間帯（5分枠）"
    ws.Cells(5, 2).Value = "時間帯最悪値"
    For di = 1 To nDays
        ws.Cells(5, 2 + di).Value = Format(dayList(di), "m/d") & _
            "（" & wdArr(Weekday(dayList(di)) - 1) & "）"
    Next di
    Call StyleHeader(ws, 5, 2 + nDays, True)
    ws.Cells(5, 2).Interior.Color = COL_DARKRED
    ws.Cells(5, 2).Font.Color = FONT_ONDARK

    ' データ行
    Dim outRow As Long
    outRow = 6
    For si = 1 To nSlots
        Dim slotStr As String
        slotStr = Format(Hour(slotTimes(si)), "00") & ":" & Format(Minute(slotTimes(si)), "00")
        ws.Cells(outRow, 1).Value = slotStr

        Dim slotWorst As Long
        slotWorst = 0
        For di = 1 To nDays
            If bucketMax(si, di) > slotWorst Then slotWorst = bucketMax(si, di)
        Next di
        ws.Cells(outRow, 2).Value = slotWorst

        For di = 1 To nDays
            ws.Cells(outRow, 2 + di).Value = bucketMax(si, di)
        Next di

        Call StyleDataRow(ws, outRow, 2 + nDays, outRow Mod 2 = 0)

        With ws.Cells(outRow, 2)
            .Font.Bold = True
            If slotWorst >= critTh And slotWorst > 0 Then
                .Interior.Color = COL_DUSTRED
                .Font.Color = RGB(140, 40, 40)
            ElseIf slotWorst >= warnTh And slotWorst > 0 Then
                .Interior.Color = COL_AMBER
                .Font.Color = FONT_ONLIGHT
            ElseIf slotWorst > 0 Then
                .Interior.Color = COL_SAGE
                .Font.Color = FONT_ONLIGHT
            End If
        End With

        For di = 1 To nDays
            Dim dv As Long
            dv = bucketMax(si, di)
            If maxAll > 0 And dv > 0 Then
                Dim ratio As Double
                ratio = dv / maxAll
                Dim r2 As Long, g2 As Long, b2 As Long
                ' セージ系グラデーション（低:薄セージ → 高:ダスティローズ）
                r2 = CLng(186 + ratio * 50)
                g2 = CLng(214 - ratio * 80)
                b2 = CLng(175 - ratio * 80)
                ws.Cells(outRow, 2 + di).Interior.Color = RGB(r2, g2, b2)
            End If
        Next di

        outRow = outRow + 1
    Next si

    ' 最大値行
    With ws.Range(ws.Cells(outRow, 1), ws.Cells(outRow, 2 + nDays))
        .Interior.Color = COL_DARK
        .Font.Color = FONT_ONDARK
        .Font.Bold = True
        .Font.Name = "Arial"
        .HorizontalAlignment = xlCenter
        .Borders.LineStyle = xlContinuous
        .Borders.Color = RGB(180, 180, 180)
    End With
    ws.Cells(outRow, 1).Value = "最大"
    ws.Cells(outRow, 2).Value = maxAll
    For di = 1 To nDays
        ws.Cells(outRow, 2 + di).Formula = "=MAX(" & _
            ws.Cells(6, 2 + di).Address(True, True) & ":" & _
            ws.Cells(outRow - 1, 2 + di).Address(True, True) & ")"
    Next di
    ws.Rows(outRow).RowHeight = 22

    ' 右端：時間帯（1時間）最悪値サマリー
    Dim summCol As Long
    summCol = 4 + nDays
    ws.Cells(5, summCol).Value = "時間帯（1時間）"
    ws.Cells(5, summCol + 1).Value = "最悪値"
    With ws.Range(ws.Cells(5, summCol), ws.Cells(5, summCol + 1))
        .Interior.Color = COL_DARKRED
        .Font.Color = FONT_ONDARK
        .Font.Bold = True: .Font.Name = "Arial"
        .HorizontalAlignment = xlCenter
        .Borders.LineStyle = xlContinuous
    End With
    ws.Columns(summCol).ColumnWidth = 18
    ws.Columns(summCol + 1).ColumnWidth = 10

    ReDim hourWorst(0 To 23) As Long
    Dim hh As Long
    For hh = 0 To 23: hourWorst(hh) = 0: Next hh
    For si = 1 To nSlots
        Dim sH As Long
        sH = Hour(slotTimes(si))
        Dim hw As Long: hw = 0
        For di = 1 To nDays
            If bucketMax(si, di) > hw Then hw = bucketMax(si, di)
        Next di
        If hw > hourWorst(sH) Then hourWorst(sH) = hw
    Next si

    Dim prevHour As Long: prevHour = -1
    Dim summRow As Long: summRow = 6
    For si = 1 To nSlots
        Dim curH As Long
        curH = Hour(slotTimes(si))
        If curH <> prevHour Then
            ws.Cells(summRow, summCol).Value = _
                Format(curH, "00") & ":00 〜 " & Format(curH, "00") & ":59"
            ws.Cells(summRow, summCol + 1).Value = hourWorst(curH)
            With ws.Cells(summRow, summCol)
                .Font.Name = "Arial": .Font.Size = 10: .Font.Bold = True
                .Font.Color = FONT_ONLIGHT
                .Interior.Color = COL_LIGHT
                .HorizontalAlignment = xlCenter
                .Borders.LineStyle = xlContinuous
                .Borders.Color = RGB(180, 180, 180)
            End With
            With ws.Cells(summRow, summCol + 1)
                .Font.Name = "Arial": .Font.Size = 10: .Font.Bold = True
                .HorizontalAlignment = xlCenter
                .Borders.LineStyle = xlContinuous
                .Borders.Color = RGB(180, 180, 180)
                If hourWorst(curH) >= critTh And hourWorst(curH) > 0 Then
                    .Interior.Color = COL_DUSTRED
                    .Font.Color = RGB(140, 40, 40)
                ElseIf hourWorst(curH) >= warnTh And hourWorst(curH) > 0 Then
                    .Interior.Color = COL_AMBER
                    .Font.Color = FONT_ONLIGHT
                Else
                    .Interior.Color = COL_SAGE
                    .Font.Color = FONT_ONLIGHT
                End If
            End With
            prevHour = curH
        End If
        summRow = summRow + 1
    Next si

    ws.Activate
    ws.Cells(6, 1).Select
    ActiveWindow.FreezePanes = False
    ActiveWindow.FreezePanes = True

    ws.Columns("A:A").ColumnWidth = 14
    ws.Columns("B:B").ColumnWidth = 14
End Sub

' ============================================================
' 処理②：受電＋折り返し分析（期間絞り込み対応）
' ============================================================
Sub Build_Callback_Analysis()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("受電＋折り返し分析")

    Dim dtStart As Date, dtEnd As Date
    If Not GetDateRange(dtStart, dtEnd) Then Exit Sub

    Dim data() As String, nRows As Long
    Call ReadInputData(data, nRows)
    If nRows = 0 Then Exit Sub

    ' 発信データインデックス
    ReDim outKeys(1 To nRows) As String
    ReDim outDTs(1 To nRows) As Date
    Dim nOut As Long: nOut = 0
    Dim i As Long
    For i = 1 To nRows
        If data(i, 1) = "発信" Then
            Dim callDT As Date
            callDT = ParseDT(data(i, 4))
            If callDT > 0 Then
                Dim callDate As Date: callDate = Int(callDT)
                If callDate >= dtStart And callDate <= dtEnd Then
                    Dim dst8 As String
                    dst8 = NormPhone(data(i, 3), NORM_DIGITS)
                    Dim dkey As String
                    dkey = dst8 & "|" & Format(callDate, "yyyy/mm/dd")
                    nOut = nOut + 1
                    outKeys(nOut) = dkey
                    outDTs(nOut) = callDT
                End If
            End If
        End If
    Next i

    ws.Rows("3:" & ws.Rows.Count).ClearContents
    ws.Rows("3:" & ws.Rows.Count).ClearFormats

    Dim outRow As Long: outRow = 3

    For i = 1 To nRows
        If data(i, 1) = "着信" Then
            Dim recvDT As Date
            recvDT = ParseDT(data(i, 4))
            If recvDT = 0 Then GoTo NextRow
            ' 期間外スキップ
            Dim recvDateD As Date: recvDateD = Int(recvDT)
            If recvDateD < dtStart Or recvDateD > dtEnd Then GoTo NextRow

            Dim ivrDT  As Date: ivrDT  = ParseDT(data(i, 5))
            Dim stDT   As Date: stDT   = ParseDT(data(i, 6))
            Dim endDT  As Date: endDT  = ParseDT(data(i, 7))
            Dim res    As String: res  = data(i, 8)
            Dim src    As String: src  = data(i, 2)
            Dim src8   As String: src8 = NormPhone(src, NORM_DIGITS)
            Dim recvDateStr As String
            recvDateStr = Format(recvDateD, "yyyy/mm/dd")

            ' 通話時間
            Dim talkMin As String: talkMin = ""
            If stDT > 0 And endDT > 0 And endDT > stDT Then
                talkMin = Format((endDT - stDT) * 1440, "0.0")
            End If

            ' 待ち時間 = G - E（秒）
            Dim waitSec As String: waitSec = "-"
            If ivrDT > 0 And endDT > 0 And endDT > ivrDT Then
                waitSec = Format((endDT - ivrDT) * 86400, "0")
            End If

            ' 折り返し検索（同日内）
            Dim cbFlag As String: cbFlag = "なし"
            Dim cbDTStr As String: cbDTStr = ""
            Dim cbMin As String: cbMin = ""
            Dim searchKey As String
            searchKey = src8 & "|" & recvDateStr
            Dim bestDT As Date: bestDT = 0
            Dim j As Long
            For j = 1 To nOut
                If outKeys(j) = searchKey Then
                    If outDTs(j) > recvDT Then
                        If bestDT = 0 Or outDTs(j) < bestDT Then
                            bestDT = outDTs(j)
                        End If
                    End If
                End If
            Next j
            If bestDT > 0 Then
                cbFlag  = "あり"
                cbDTStr = Format(bestDT, "yyyy/mm/dd hh:mm")
                cbMin   = Format((bestDT - recvDT) * 1440, "0.0")
            End If

            ' 短時間通話フラグ（型不一致修正済み）
            Dim shortFlag As String: shortFlag = ""
            If talkMin <> "" And IsNumeric(talkMin) Then
                If CDbl(talkMin) < 1 Then shortFlag = "○"
            End If

            Call StyleDataRow(ws, outRow, 10, outRow Mod 2 = 0)
            ws.Cells(outRow, 1).Value = Format(recvDT, "yyyy/mm/dd hh:mm")
            ws.Cells(outRow, 2).Value = src
            ws.Cells(outRow, 3).Value = src8
            ws.Cells(outRow, 4).Value = res
            ws.Cells(outRow, 5).Value = talkMin
            ws.Cells(outRow, 6).Value = waitSec
            ws.Cells(outRow, 7).Value = cbFlag
            ws.Cells(outRow, 8).Value = cbDTStr
            ws.Cells(outRow, 9).Value = cbMin
            ws.Cells(outRow, 10).Value = shortFlag

            With ws.Cells(outRow, 7)
                If cbFlag = "あり" Then
                    .Interior.Color = COL_SAGE
                    .Font.Bold = True
                    .Font.Color = RGB(40, 80, 40)
                Else
                    .Interior.Color = COL_AMBER
                    .Font.Color = FONT_ONLIGHT
                End If
            End With

            outRow = outRow + 1
        End If
NextRow:
    Next i
End Sub

' ============================================================
' 処理③：時間帯（1時間単位）×折り返し分析（期間絞り込み対応）
' ============================================================
Sub Build_5Min_Callback()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("5分×折り返し分析")
    ws.Rows("3:" & ws.Rows.Count).ClearContents
    ws.Rows("3:" & ws.Rows.Count).ClearFormats

    ' ヘッダーを更新
    ws.Cells(1, 1).Value = "時間帯別 折り返し分析 ― マクロ実行で更新"
    ws.Range("A2:G2").ClearContents
    Dim hdrLabels(0 To 6) As String
    hdrLabels(0) = "時間帯（1時間）"
    hdrLabels(1) = "受電数"
    hdrLabels(2) = "折り返し数"
    hdrLabels(3) = "折り返し率"
    hdrLabels(4) = "平均折り返し時間(分)"
    hdrLabels(5) = "中央値(分)"
    hdrLabels(6) = "90%tile(分)"
    Dim hc As Long
    For hc = 0 To 6
        ws.Cells(2, hc + 1).Value = hdrLabels(hc)
    Next hc
    Call StyleHeader(ws, 2, 7, True)

    Dim dtStart As Date, dtEnd As Date
    If Not GetDateRange(dtStart, dtEnd) Then Exit Sub

    Dim wsCB As Worksheet
    Set wsCB = ThisWorkbook.Sheets("受電＋折り返し分析")
    Dim lastCB As Long
    lastCB = wsCB.Cells(wsCB.Rows.Count, 1).End(xlUp).Row
    If lastCB < 3 Then Exit Sub

    ' 時間帯（0〜23）ごとに集計
    ReDim hRecv(0 To 23)  As Long
    ReDim hCB(0 To 23)    As Long
    ReDim hTimes(0 To 23, 1 To 5000) As Double
    ReDim hTimeCnt(0 To 23) As Long

    Dim r As Long
    For r = 3 To lastCB
        Dim dtStr As String
        dtStr = CStr(wsCB.Cells(r, 1).Value)
        If dtStr = "" Or Not IsDate(dtStr) Then GoTo NextCBRow
        Dim dt As Date: dt = CDate(dtStr)
        ' 期間チェック
        Dim dtD As Date: dtD = Int(dt)
        If dtD < dtStart Or dtD > dtEnd Then GoTo NextCBRow

        Dim hh As Long: hh = Hour(dt)
        hRecv(hh) = hRecv(hh) + 1

        If CStr(wsCB.Cells(r, 7).Value) = "あり" Then
            hCB(hh) = hCB(hh) + 1
            Dim cbMStr As String
            cbMStr = CStr(wsCB.Cells(r, 9).Value)
            If cbMStr <> "" And IsNumeric(cbMStr) Then
                hTimeCnt(hh) = hTimeCnt(hh) + 1
                hTimes(hh, hTimeCnt(hh)) = CDbl(cbMStr)
            End If
        End If
NextCBRow:
    Next r

    ' 出力
    Dim outRow As Long: outRow = 3
    Dim startH As Long, endH As Long
    startH = Hour(TimeValue(TIME_START))
    endH   = Hour(TimeValue(TIME_END))

    For hh = startH To endH
        If hRecv(hh) = 0 Then GoTo NextHour
        Dim total As Long: total = hRecv(hh)
        Dim cbCnt As Long: cbCnt = hCB(hh)
        Dim rate As Double
        If total > 0 Then rate = cbCnt / total * 100 Else rate = 0

        Dim nT As Long: nT = hTimeCnt(hh)
        Dim avgT As String: avgT = ""
        Dim medT As String: medT = ""
        Dim p90T As String: p90T = ""

        If nT > 0 Then
            ReDim times(1 To nT) As Double
            Dim k As Long
            Dim sm As Double: sm = 0
            For k = 1 To nT
                times(k) = hTimes(hh, k)
                sm = sm + times(k)
            Next k
            avgT = Format(sm / nT, "0.0")
            ' ソート
            Dim ti As Long, tj As Long, tmp As Double
            For ti = 1 To nT - 1
                For tj = 1 To nT - ti
                    If times(tj) > times(tj + 1) Then
                        tmp = times(tj): times(tj) = times(tj + 1): times(tj + 1) = tmp
                    End If
                Next tj
            Next ti
            If nT Mod 2 = 1 Then
                medT = Format(times((nT + 1) \ 2), "0.0")
            Else
                medT = Format((times(nT \ 2) + times(nT \ 2 + 1)) / 2, "0.0")
            End If
            Dim p90idx As Long
            p90idx = CLng(nT * 0.9)
            If p90idx < 1 Then p90idx = 1
            If p90idx > nT Then p90idx = nT
            p90T = Format(times(p90idx), "0.0")
        End If

        Call StyleDataRow(ws, outRow, 7, outRow Mod 2 = 0)
        ws.Cells(outRow, 1).Value = Format(hh, "00") & ":00 〜 " & Format(hh, "00") & ":59"
        ws.Cells(outRow, 2).Value = total
        ws.Cells(outRow, 3).Value = cbCnt
        ws.Cells(outRow, 4).Value = Format(rate, "0.0") & "%"
        ws.Cells(outRow, 5).Value = avgT
        ws.Cells(outRow, 6).Value = medT
        ws.Cells(outRow, 7).Value = p90T

        With ws.Cells(outRow, 4)
            If total > 0 Then
                If rate >= 50 Then
                    .Interior.Color = COL_DUSTRED
                    .Font.Color = RGB(140, 40, 40)
                ElseIf rate >= 30 Then
                    .Interior.Color = COL_AMBER
                    .Font.Color = FONT_ONLIGHT
                Else
                    .Interior.Color = COL_SAGE
                    .Font.Color = FONT_ONLIGHT
                End If
            End If
        End With

        outRow = outRow + 1
NextHour:
    Next hh
End Sub

' ============================================================
' 処理④：集計シート（期間絞り込み対応）
' ============================================================
Sub Build_Aggregate()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("集計")
    ws.Rows("3:" & ws.Rows.Count).ClearContents
    ws.Rows("3:" & ws.Rows.Count).ClearFormats

    Dim dtStart As Date, dtEnd As Date
    If Not GetDateRange(dtStart, dtEnd) Then Exit Sub

    Dim data() As String, nRows As Long
    Call ReadInputData(data, nRows)
    If nRows = 0 Then Exit Sub

    ' 日付リスト（期間内のみ）
    ReDim dateKeys(1 To 500) As String
    Dim nDates As Long: nDates = 0
    Dim i As Long, j As Long
    For i = 1 To nRows
        If data(i, 1) = "着信" Then
            Dim dt As Date: dt = ParseDT(data(i, 4))
            If dt > 0 Then
                Dim dtD As Date: dtD = Int(dt)
                If dtD >= dtStart And dtD <= dtEnd Then
                    Dim dk As String: dk = Format(dtD, "yyyy/mm/dd")
                    Dim found As Boolean: found = False
                    For j = 1 To nDates
                        If dateKeys(j) = dk Then found = True: Exit For
                    Next j
                    If Not found Then
                        nDates = nDates + 1: dateKeys(nDates) = dk
                    End If
                End If
            End If
        End If
    Next i
    If nDates = 0 Then Exit Sub

    ' 日付ソート
    Dim ti As Long, tja As Long, tmpS As String
    For ti = 1 To nDates - 1
        For tja = 1 To nDates - ti
            If dateKeys(tja) > dateKeys(tja + 1) Then
                tmpS = dateKeys(tja): dateKeys(tja) = dateKeys(tja + 1): dateKeys(tja + 1) = tmpS
            End If
        Next tja
    Next ti

    ' 結果種別
    Dim results(0 To 5) As String
    results(0) = "応答": results(1) = "放棄呼": results(2) = "話中"
    results(3) = "未応答": results(4) = "接続異常": results(5) = Chr(34)
    Dim nRes As Long: nRes = 5

    ReDim cross(1 To 500, 0 To 5) As Long
    ReDim dailyIn(1 To 500)    As Long
    ReDim dailyOut(1 To 500)   As Long
    ReDim dailyAband(1 To 500) As Long
    ReDim dailyAnsw(1 To 500)  As Long

    For i = 1 To nRows
        Dim knd As String: knd = data(i, 1)
        Dim dt2 As Date: dt2 = ParseDT(data(i, 4))
        If dt2 = 0 Then GoTo NextRow2
        Dim dtD2 As Date: dtD2 = Int(dt2)
        If dtD2 < dtStart Or dtD2 > dtEnd Then GoTo NextRow2
        Dim dk2 As String: dk2 = Format(dtD2, "yyyy/mm/dd")
        Dim di As Long: di = 0
        For j = 1 To nDates
            If dateKeys(j) = dk2 Then di = j: Exit For
        Next j
        If di = 0 Then GoTo NextRow2

        If knd = "着信" Then
            Dim res2 As String: res2 = data(i, 8)
            dailyIn(di) = dailyIn(di) + 1
            Dim ri As Long
            For ri = 0 To nRes
                If res2 = results(ri) Then
                    cross(di, ri) = cross(di, ri) + 1: Exit For
                End If
            Next ri
            If res2 = "応答" Then dailyAnsw(di) = dailyAnsw(di) + 1
            If res2 = "放棄呼" Then dailyAband(di) = dailyAband(di) + 1
        ElseIf knd = "発信" Then
            dailyOut(di) = dailyOut(di) + 1
        End If
NextRow2:
    Next i

    Dim outRow As Long: outRow = 3

    ' ① クロス集計
    Call WriteSectionTitle(ws, outRow, 1, 3 + nDates, _
        "① 呼び出し結果別 × 日付 クロス集計（着信のみ）", COL_DARK, True)
    outRow = outRow + 1
    ws.Cells(outRow, 2).Value = "日付"
    ws.Cells(outRow, 2).Interior.Color = COL_LIGHT
    ws.Cells(outRow, 2).Font.Bold = True
    ws.Cells(outRow, 2).Font.Color = FONT_ONLIGHT
    ws.Cells(outRow, 2).Borders.LineStyle = xlContinuous
    ws.Cells(outRow, 2).HorizontalAlignment = xlCenter
    For ri = 0 To nRes: ws.Cells(outRow, 3 + ri).Value = results(ri): Next ri
    ws.Cells(outRow, 3 + nRes + 1).Value = "総計"
    Call StyleHeader(ws, outRow, 3 + nRes + 1, False)
    outRow = outRow + 1
    For di = 1 To nDates
        Call StyleDataRow(ws, outRow, 3 + nRes + 1, di Mod 2 = 0)
        ws.Cells(outRow, 2).Value = dateKeys(di)
        Dim rowTotal As Long: rowTotal = 0
        For ri = 0 To nRes
            ws.Cells(outRow, 3 + ri).Value = cross(di, ri)
            rowTotal = rowTotal + cross(di, ri)
        Next ri
        ws.Cells(outRow, 3 + nRes + 1).Value = rowTotal
        outRow = outRow + 1
    Next di
    outRow = outRow + 1

    ' ② 日別サマリー
    Call WriteSectionTitle(ws, outRow, 1, 3 + nDates, _
        "② 日別対応サマリー（①入電 ②受電 ③放棄呼 ④発信 ⑤全体）", COL_DARK, True)
    outRow = outRow + 1
    ws.Cells(outRow, 2).Value = "項目"
    ws.Cells(outRow, 2).Interior.Color = COL_LIGHT
    ws.Cells(outRow, 2).Font.Bold = True
    ws.Cells(outRow, 2).Font.Color = FONT_ONLIGHT
    ws.Cells(outRow, 2).Borders.LineStyle = xlContinuous
    ws.Cells(outRow, 2).HorizontalAlignment = xlCenter
    For di = 1 To nDates: ws.Cells(outRow, 2 + di).Value = dateKeys(di): Next di
    ws.Cells(outRow, 3 + nDates).Value = "合計"
    Call StyleHeader(ws, outRow, 3 + nDates, False)
    outRow = outRow + 1

    Dim kpiLabels(0 To 5) As String
    kpiLabels(0) = "①入電件数": kpiLabels(1) = "②受電件数（応答）"
    kpiLabels(2) = "③放棄呼数": kpiLabels(3) = "（参考）呼損率"
    kpiLabels(4) = "④発信件数": kpiLabels(5) = "⑤全体対応（②＋④）"

    Dim ki As Long
    For ki = 0 To 5
        Call StyleDataRow(ws, outRow, 3 + nDates, ki Mod 2 = 0)
        ws.Cells(outRow, 2).Value = kpiLabels(ki)
        ws.Cells(outRow, 2).Font.Bold = True
        ws.Cells(outRow, 2).HorizontalAlignment = xlLeft
        Dim total2 As Long: total2 = 0
        For di = 1 To nDates
            Dim v As Variant
            Select Case ki
                Case 0: v = dailyIn(di)
                Case 1: v = dailyAnsw(di)
                Case 2: v = dailyAband(di)
                Case 3:
                    If dailyIn(di) > 0 Then
                        v = Format(dailyAband(di) / dailyIn(di) * 100, "0.0") & "%"
                    Else: v = "-"
                    End If
                Case 4: v = dailyOut(di)
                Case 5: v = dailyAnsw(di) + dailyOut(di)
            End Select
            ws.Cells(outRow, 2 + di).Value = v
            If IsNumeric(v) Then total2 = total2 + CLng(v)
        Next di
        If ki = 3 Then
            ws.Cells(outRow, 3 + nDates).Value = "-"
        Else
            ws.Cells(outRow, 3 + nDates).Value = total2
            ws.Cells(outRow, 3 + nDates).Interior.Color = COL_AMBER
            ws.Cells(outRow, 3 + nDates).Font.Bold = True
        End If
        outRow = outRow + 1
    Next ki
    outRow = outRow + 1

    ' ③ 時間帯別待ち時間
    Call WriteSectionTitle(ws, outRow, 1, 2 + nDates, _
        "③ 時間帯別 待ち時間（秒）平均　※G列－E列。E列が " & Chr(34) & " の場合は除外", _
        COL_DARK, True)
    outRow = outRow + 1

    ws.Cells(outRow, 2).Interior.Color = COL_LIGHT
    ws.Cells(outRow, 2).Font.Bold = True
    ws.Cells(outRow, 2).Font.Color = FONT_ONLIGHT

    For si = 1 To nSlots: slotTimes2(si) = tt: tt = tt + TimeSerial(0, 5, 0): Next si

                Dim recvD3 As Date: recvD3 = Int(recvDT3)

                Dim fsi2 As Long: fsi2 = 0
                For si = 1 To nSlots

            For di4 = 1 To nDates

                    ws.Cells(outRow, 2 + di4).Value = Format(waitSum(si2, di4) / waitCnt(si2, di4), "0")
                Else
                    ws.Cells(outRow, 2 + di4).Value = "-"
                End If
            Next di4
            outRow = outRow + 1
NextSlot:
        Next si2

SkipWaitSection:

    Call WriteSectionTitle(ws, outRow, 1, 5, _
        "④ 日付別 × 時間帯（午前/午後）別 着信件数", COL_DARK, True)

    ReDim pmCnt(1 To 500) As Long

    ' ⑤ 放棄呼 待ち時間一覧
    Call WriteSectionTitle(ws, outRow, 1, 10, _
        "⑤ 着信・放棄呼 日付別 待ち時間一覧（1件1列、単位：秒、G列－E列）", COL_DARK, True)
    outRow = outRow + 1
    ws.Cells(outRow, 2).Value = "日付"
    ws.Cells(outRow, 3).Value = "着信件数"
    ws.Cells(outRow, 4).Value = "放棄呼件数"
    ws.Cells(outRow, 5).Value = "待ち時間①（以降1件1列）"
    Dim cc As Long
    For cc = 2 To 5
        ws.Cells(outRow, cc).Interior.Color = COL_LIGHT
        ws.Cells(outRow, cc).Font.Bold = True
        ws.Cells(outRow, cc).Font.Color = FONT_ONLIGHT
        ws.Cells(outRow, cc).Font.Name = "Arial"
        ws.Cells(outRow, cc).HorizontalAlignment = xlCenter
        ws.Cells(outRow, cc).Borders.LineStyle = xlContinuous
        ws.Cells(outRow, cc).Borders.Color = RGB(180, 180, 180)
    Next cc
    ws.Rows(outRow).RowHeight = 22
    outRow = outRow + 1

    For di = 1 To nDates
        Dim inbCnt As Long: inbCnt = 0
        Dim abdCnt As Long: abdCnt = 0
        ReDim waitList(1 To nRows) As Double
        Dim nWait As Long: nWait = 0

        For i = 1 To nRows
            If data(i, 1) = "着信" Then
                Dim dt5 As Date: dt5 = ParseDT(data(i, 4))
                If dt5 = 0 Then GoTo NextRow5
                If Format(Int(dt5), "yyyy/mm/dd") <> dateKeys(di) Then GoTo NextRow5
                inbCnt = inbCnt + 1
                If data(i, 8) = "放棄呼" Then
                    abdCnt = abdCnt + 1
                    Dim ivrW As String: ivrW = data(i, 5)
                    Dim endW As String: endW = data(i, 7)
                    If ivrW <> Chr(34) And ivrW <> "" Then
                        Dim ivrD As Date: ivrD = ParseDT(ivrW)
                        Dim endD As Date: endD = ParseDT(endW)
                        If ivrD > 0 And endD > ivrD Then
                            nWait = nWait + 1
                            waitList(nWait) = (endD - ivrD) * 86400
                        End If
                    End If
                End If
            End If
NextRow5:
        Next i

        Dim nOutCols As Long: nOutCols = 4 + nWait
        If nOutCols < 4 Then nOutCols = 4
        Call StyleDataRow(ws, outRow, nOutCols, di Mod 2 = 0)
        ws.Cells(outRow, 2).Value = dateKeys(di)
        ws.Cells(outRow, 3).Value = inbCnt
        ws.Cells(outRow, 4).Value = abdCnt
        Dim wc As Long
        For wc = 1 To nWait
            ws.Cells(outRow, 4 + wc).Value = Format(waitList(wc), "0")
        Next wc
        outRow = outRow + 1
    Next di

    ws.Columns("B:B").ColumnWidth = 16
    ws.Columns("C:C").ColumnWidth = 14
    ws.Columns("D:D").ColumnWidth = 14
End Sub

' ============================================================
' 処理⑤：傾向分析サマリー（期間絞り込み対応）
' ============================================================
Sub Build_Summary()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("傾向分析サマリー")
    ws.Rows("3:" & ws.Rows.Count).ClearContents
    ws.Rows("3:" & ws.Rows.Count).ClearFormats

    Dim dtStart As Date, dtEnd As Date
    If Not GetDateRange(dtStart, dtEnd) Then Exit Sub

    Dim wsCB As Worksheet
    Set wsCB = ThisWorkbook.Sheets("受電＋折り返し分析")
    Dim lastR As Long
    lastR = wsCB.Cells(wsCB.Rows.Count, 1).End(xlUp).Row
    If lastR < 3 Then Exit Sub

    Dim totalRecv As Long: totalRecv = lastR - 2
    Dim totalCB As Long: totalCB = 0
    ReDim cbTimes(1 To totalRecv) As Double
    Dim nCBT As Long: nCBT = 0
    Dim totalShort As Long: totalShort = 0

    Dim r As Long
    For r = 3 To lastR
        If CStr(wsCB.Cells(r, 7).Value) = "あり" Then
            totalCB = totalCB + 1
            Dim cm As String: cm = CStr(wsCB.Cells(r, 9).Value)
            If cm <> "" And IsNumeric(cm) Then
                nCBT = nCBT + 1: cbTimes(nCBT) = CDbl(cm)
            End If
        End If
        If CStr(wsCB.Cells(r, 10).Value) = "○" Then totalShort = totalShort + 1
    Next r

    Dim wsEV As Worksheet
    Set wsEV = ThisWorkbook.Sheets("イベントテーブル")
    Dim maxConc As Long: maxConc = 0
    Dim lastEVR As Long
    lastEVR = wsEV.Cells(wsEV.Rows.Count, 2).End(xlUp).Row
    Dim ev As Long
    For ev = 6 To lastEVR
        Dim evVal As Variant: evVal = wsEV.Cells(ev, 2).Value
        If IsNumeric(evVal) And CLng(evVal) > maxConc Then maxConc = CLng(evVal)
    Next ev

    Dim avgCB As Double: avgCB = 0
    Dim medCB As Double: medCB = 0
    If nCBT > 0 Then
        Dim sm As Double: sm = 0
        Dim k As Long
        For k = 1 To nCBT: sm = sm + cbTimes(k): Next k
        avgCB = sm / nCBT
        Dim ti As Long, tj As Long, tmp As Double
        For ti = 1 To nCBT - 1
            For tj = 1 To nCBT - ti
                If cbTimes(tj) > cbTimes(tj + 1) Then
                    tmp = cbTimes(tj): cbTimes(tj) = cbTimes(tj + 1): cbTimes(tj + 1) = tmp
                End If
            Next tj
        Next ti
        If nCBT Mod 2 = 1 Then
            medCB = cbTimes((nCBT + 1) \ 2)
        Else
            medCB = (cbTimes(nCBT \ 2) + cbTimes(nCBT \ 2 + 1)) / 2
        End If
    End If

    Dim cbRate As Double
    If totalRecv > 0 Then cbRate = totalCB / totalRecv * 100 Else cbRate = 0

    Dim outRow As Long: outRow = 3
    Call WriteKPI(ws, outRow, 2, "総着信数", CStr(totalRecv), "集計期間合計", COL_DARK)
    Call WriteKPI(ws, outRow, 4, "最大同時通話数", CStr(maxConc), "必要回線数ピーク", COL_DARKRED)
    outRow = outRow + 4
    Call WriteKPI(ws, outRow, 2, "折り返し件数", CStr(totalCB), _
        "折り返し率 " & Format(cbRate, "0.0") & "%", COL_MID)
    Call WriteKPI(ws, outRow, 4, "平均折り返し時間", _
        Format(avgCB, "0.0") & "分", "中央値 " & Format(medCB, "0.0") & "分", RGB(70, 110, 70))
    outRow = outRow + 4

    Call WriteSectionTitle(ws, outRow, 2, 5, "🔍 ボトルネック診断", COL_DARKRED, True)
    outRow = outRow + 1

    Dim diagItems(0 To 3) As String
    Dim diagVals(0 To 3) As String
    diagItems(0) = "最大同時通話数（回線ピーク）"
    diagVals(0) = CStr(maxConc) & " 回線"
    diagItems(1) = "折り返し率"
    diagVals(1) = Format(cbRate, "0.0") & "% （" & totalCB & "/" & totalRecv & "件）"
    diagItems(2) = "平均折り返し遅延"
    diagVals(2) = Format(avgCB, "0.0") & "分（中央値 " & Format(medCB, "0.0") & "分）"
    diagItems(3) = "短時間通話（<1分）件数"
    diagVals(3) = CStr(totalShort) & " 件"

    Dim ki As Long
    For ki = 0 To 3
        ws.Cells(outRow, 2).Value = diagItems(ki)
        ws.Cells(outRow, 2).Font.Name = "Arial": ws.Cells(outRow, 2).Font.Bold = True
        ws.Cells(outRow, 2).Font.Color = FONT_ONLIGHT
        ws.Cells(outRow, 2).Interior.Color = COL_LGRAY
        ws.Cells(outRow, 2).HorizontalAlignment = xlLeft
        ws.Cells(outRow, 2).Borders.LineStyle = xlContinuous
        ws.Cells(outRow, 2).Borders.Color = RGB(180, 180, 180)
        On Error Resume Next
        ws.Range(ws.Cells(outRow, 3), ws.Cells(outRow, 5)).Merge
        On Error GoTo 0
        ws.Cells(outRow, 3).Value = diagVals(ki)
        ws.Cells(outRow, 3).Font.Name = "Arial"
        ws.Cells(outRow, 3).Font.Color = FONT_ONLIGHT
        ws.Cells(outRow, 3).Interior.Color = COL_CREAM
        ws.Cells(outRow, 3).HorizontalAlignment = xlLeft
        ws.Cells(outRow, 3).Borders.LineStyle = xlContinuous
        ws.Cells(outRow, 3).Borders.Color = RGB(180, 180, 180)
        ws.Rows(outRow).RowHeight = 20
        outRow = outRow + 1
    Next ki

    ws.Columns("B:B").ColumnWidth = 30
    ws.Columns("C:D").ColumnWidth = 18
End Sub

' デバッグ確認用（問題解決後に削除可）
' ============================================================
Sub デバッグ確認()
    Dim data() As String, nRows As Long
    Call ReadInputData(data, nRows)

    Dim msg As String
    msg = "総行数: " & nRows & vbCrLf

    If nRows = 0 Then
        MsgBox "データが読み込めていません。", vbExclamation
        Exit Sub
    End If

    Dim i As Long
    For i = 1 To 3
        If i > nRows Then Exit For
        msg = msg & "---行" & i & "---" & vbCrLf
        msg = msg & "A(種別)=[" & data(i, 1) & "]" & vbCrLf
        msg = msg & "D(受電日時)=[" & data(i, 4) & "]" & vbCrLf
        msg = msg & "H(結果)=[" & data(i, 8) & "]" & vbCrLf
    Next i

    Dim nRecv As Long, nSend As Long, nOther As Long
    For i = 1 To nRows
        Select Case data(i, 1)
            Case "着信": nRecv = nRecv + 1
            Case "発信": nSend = nSend + 1
            Case Else:   nOther = nOther + 1
        End Select
    Next i
    msg = msg & "---集計---" & vbCrLf
    msg = msg & "着信件数: " & nRecv & vbCrLf
    msg = msg & "発信件数: " & nSend & vbCrLf
    msg = msg & "未分類: " & nOther & vbCrLf

    Dim dt As Date
    dt = ParseDT(data(1, 4))
    msg = msg & "1行目の日時パース結果: [" & CStr(dt) & "]" & vbCrLf

    MsgBox msg, vbInformation, "デバッグ確認"
End Sub
