' ════════════════════════════════════════════
' ① CSV 取り込み
' ════════════════════════════════════════════
Sub ImportCSV()
    Dim fPath As String
    fPath = Application.GetOpenFilename("CSVファイル,*.csv", , "CSVを選択してください")
    If fPath = "False" Then Exit Sub

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets(DATA_SHEET)
    ws.Cells.ClearContents
    ws.Cells.ClearFormats

    ' ファイルをバイナリで丸ごと読み込み
    Dim fileNo As Integer: fileNo = FreeFile
    Dim buf As String
    Open fPath For Binary As #fileNo
    buf = Space(LOF(fileNo))
    Get #fileNo, , buf
    Close #fileNo

    ' ダブルクォート内の改行をスペースに置換
    Dim inQ As Boolean: inQ = False
    Dim cleanBuf As String: cleanBuf = ""
    Dim i As Long
    For i = 1 To Len(buf)
        Dim ch As String: ch = Mid(buf, i, 1)
        If ch = Chr(34) Then inQ = Not inQ
        If (ch = vbLf Or ch = vbCr) And inQ Then
            cleanBuf = cleanBuf & " "
        Else
            cleanBuf = cleanBuf & ch
        End If
    Next i

    ' 改行コードを統一してLFに
    cleanBuf = Replace(cleanBuf, vbCrLf, vbLf)
    cleanBuf = Replace(cleanBuf, vbCr, vbLf)

    ' 行に分割
    Dim lines() As String
    lines = Split(cleanBuf, vbLf)

    Dim ln As Long: ln = 1
    Dim li As Long
    For li = 0 To UBound(lines)
        Dim lineStr As String: lineStr = Trim(lines(li))
        If lineStr = "" Then GoTo NextLine
        Dim cols() As String
        cols = SplitCSV(lineStr)
        Dim j As Integer
        For j = 0 To UBound(cols)
            ws.Cells(ln, j + 1).NumberFormat = "@"
            ws.Cells(ln, j + 1).Value = Trim(cols(j))
        Next j
        ln = ln + 1
NextLine:
    Next li

    With ws.Rows(1)
        .Font.Bold = True
        .Font.Name = "Meiryo UI"
        .Font.Color = RGB(255, 255, 255)
        .Interior.Color = RGB(31, 56, 100)
        .HorizontalAlignment = xlCenter
        .RowHeight = 28
    End With
    ws.Rows(1).AutoFilter
    ws.Columns("A:W").AutoFit
    MsgBox "CSV取り込み完了！  " & (ln - 2) & " 件", vbInformation
End Sub

' カンマ区切り分割（ダブルクォート対応）
Function SplitCSV(s As String) As String()
    Dim res() As String
    res = Split(s, ",")
    Dim i As Integer
    For i = 0 To UBound(res)
        res(i) = Replace(res(i), Chr(34), "")
    Next i
    SplitCSV = res
End Function

' ════════════════════════════════════════════
' ② 全シート更新
' ════════════════════════════════════════════
Sub UpdateAllSheets()
    Dim pws As Worksheet: Set pws = ThisWorkbook.Sheets(PANEL_SHEET)
    Dim dtStart As Date, dtEnd As Date

    On Error GoTo NoPeriod
    dtStart = CDate(pws.Cells(8, 3).Value)
    dtEnd   = CDate(pws.Cells(9, 3).Value)
    On Error GoTo 0

    If dtStart > dtEnd Then
        MsgBox "開始日が終了日より後になっています。", vbExclamation
        Exit Sub
    End If

    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    Call BuildHistorySheet(dtStart, dtEnd)
    Call BuildCatDetailSheet(dtStart, dtEnd)
    Call BuildCatWeekSheet(dtStart, dtEnd)
    Call BuildBureauSheet(dtStart, dtEnd)
    Call BuildStaffSheet(dtStart, dtEnd)

    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True
    MsgBox "全シート更新完了！" & vbCrLf & _
           "集計期間：" & Format(dtStart, "yyyy/m/d") & " ～ " & Format(dtEnd, "yyyy/m/d"), vbInformation
    Exit Sub

NoPeriod:
    On Error GoTo 0
    MsgBox "操作パネルに開始日・終了日を入力してください。", vbExclamation
End Sub

' ════════════════════════════════════════════
' 共通ヘルパー
' ════════════════════════════════════════════
Function ColIdx(ws As Worksheet, hdr As String) As Long
    Dim c As Range
    For Each c In ws.Rows(1).Cells
        If Trim(c.Value) = hdr Then ColIdx = c.Column: Exit Function
    Next c
    ColIdx = 0
End Function

Function ToDate(v As Variant) As Date
    On Error GoTo bad
    Dim s As String: s = Trim(CStr(v))
    If s = "" Or s = "Empty" Then GoTo bad
    If InStr(s, " ") > 0 Then s = Left(s, InStr(s, " ") - 1)
    ToDate = CDate(s)
    Exit Function
bad:
    ToDate = CDate("1900/1/1")
End Function

Function IsValidDate(v As Variant) As Boolean
    IsValidDate = (ToDate(v) <> CDate("1900/1/1"))
End Function

Function DateLabel(d As Date) As String
    DateLabel = Month(d) & "/" & Day(d)
End Function

Function WeekMon(d As Date) As Date
    WeekMon = d - (Weekday(d, vbMonday) - 1)
End Function

Sub StyleHeader(ws As Worksheet, r As Long, lastC As Long)
    With ws.Range(ws.Cells(r, 1), ws.Cells(r, lastC))
        .Font.Bold = True: .Font.Name = "Meiryo UI": .Font.Size = 10
        .Font.Color = RGB(255, 255, 255)
        .Interior.Color = RGB(31, 56, 100)
        .HorizontalAlignment = xlCenter: .VerticalAlignment = xlCenter
        .WrapText = True: .RowHeight = 28
        .Borders.LineStyle = xlContinuous
        .Borders.Color = RGB(189, 215, 238)
    End With
End Sub

Sub StyleData(ws As Worksheet, r1 As Long, r2 As Long, c1 As Long, c2 As Long)
    Dim r As Long
    For r = r1 To r2
        With ws.Range(ws.Cells(r, c1), ws.Cells(r, c2))
            .Interior.Color = IIf(r Mod 2 = 0, RGB(235, 242, 250), RGB(255, 255, 255))
            .Font.Name = "Meiryo UI": .Font.Size = 10
            .VerticalAlignment = xlCenter
            .Borders.LineStyle = xlContinuous
            .Borders.Color = RGB(189, 215, 238)
        End With
    Next r
End Sub

Sub StyleTotal(ws As Worksheet, r As Long, c1 As Long, c2 As Long)
    With ws.Range(ws.Cells(r, c1), ws.Cells(r, c2))
        .Font.Bold = True: .Font.Name = "Meiryo UI"
        .Interior.Color = RGB(31, 56, 100)
        .Font.Color = RGB(255, 255, 255)
        .Borders.LineStyle = xlContinuous
        .Borders.Color = RGB(100, 100, 100)
    End With
End Sub

Sub StyleTitle(ws As Worksheet, title As String, dtS As Date, dtE As Date)
    ws.Cells(1, 1).Value = title & "　（" & Format(dtS, "yyyy/m/d") & " ～ " & Format(dtE, "yyyy/m/d") & "）"
    With ws.Cells(1, 1).Font
        .Bold = True: .Size = 12: .Name = "Meiryo UI": .Color = RGB(31, 56, 100)
    End With
    ws.Rows(1).RowHeight = 24
End Sub

Sub AutoFitSheet(ws As Worksheet)
    ws.Columns.AutoFit
    Dim c As Range
    For Each c In ws.UsedRange.Columns
        If c.ColumnWidth > 20 Then c.ColumnWidth = 20
        If c.ColumnWidth < 5  Then c.ColumnWidth = 5
    Next c
    ws.Columns(1).ColumnWidth = 20
End Sub

Sub ColorDiff(c As Range, df As Long)
    If df >= 5 Then
        c.Interior.Color = RGB(198, 239, 206): c.Font.Color = RGB(0, 97, 0): c.Font.Bold = True
    ElseIf df <= -5 Then
        c.Interior.Color = RGB(255, 199, 206): c.Font.Color = RGB(156, 0, 6): c.Font.Bold = True
    Else
        c.Interior.Color = RGB(255, 235, 156): c.Font.Color = RGB(156, 87, 0)
    End If
End Sub

' ════════════════════════════════════════════
' ③ 新規履歴件数
' ════════════════════════════════════════════
Sub BuildHistorySheet(dtS As Date, dtE As Date)
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_HISTORY)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    If colD = 0 Then Exit Sub

    Dim dc As Object: Set dc = CreateObject("Scripting.Dictionary")
    Dim dateOrder As New Collection
    Dim r As Long
    For r = 2 To lastRow
        If Not IsValidDate(dws.Cells(r, colD).Value) Then GoTo Skip1
        Dim d As Date: d = ToDate(dws.Cells(r, colD).Value)
        If d < dtS Or d > dtE Then GoTo Skip1
        Dim lbl As String: lbl = DateLabel(d)
        If Not dc.Exists(lbl) Then
            dc.Add lbl, 1: dateOrder.Add lbl
        Else
            dc(lbl) = dc(lbl) + 1
        End If
Skip1:
    Next r

    StyleTitle ws, "新規履歴件数", dtS, dtE
    ws.Cells(2, 1).Value = "区分"
    Dim col As Long: col = 2
    Dim k As Variant
    For Each k In dateOrder
        ws.Cells(2, col).Value = k: col = col + 1
    Next k
    Dim tcol As Long: tcol = col
    ws.Cells(2, tcol).Value = "合計"
    StyleHeader ws, 2, tcol

    ws.Cells(3, 1).Value = "件数"
    col = 2
    For Each k In dateOrder
        ws.Cells(3, col).Value = dc(k): col = col + 1
    Next k
    ws.Cells(3, tcol).Value = "=SUM(" & ws.Cells(3, 2).Address(False, False) & ":" & ws.Cells(3, tcol - 1).Address(False, False) & ")"
    StyleData ws, 3, 3, 1, tcol
    With ws.Cells(3, 1)
        .Interior.Color = RGB(214, 228, 247): .Font.Bold = True: .HorizontalAlignment = xlCenter
    End With
    AutoFitSheet ws
End Sub

' ════════════════════════════════════════════
' ④ カテゴリー別詳細
' ════════════════════════════════════════════
Sub BuildCatDetailSheet(dtS As Date, dtE As Date)
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_CAT_DTL)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    Dim colC As Long: colC = ColIdx(dws, "業務分類詳細")
    If colD = 0 Or colC = 0 Then Exit Sub

    Dim counts   As Object: Set counts   = CreateObject("Scripting.Dictionary")
    Dim bigOrder As New Collection
    Dim bigSet   As Object: Set bigSet   = CreateObject("Scripting.Dictionary")
    Dim smallOrder As New Collection
    Dim smallSet   As Object: Set smallSet = CreateObject("Scripting.Dictionary")
    Dim dateOrder As New Collection
    Dim dateSet   As Object: Set dateSet  = CreateObject("Scripting.Dictionary")

    Dim r As Long
    For r = 2 To lastRow
        If Not IsValidDate(dws.Cells(r, colD).Value) Then GoTo Skip2
        Dim d As Date: d = ToDate(dws.Cells(r, colD).Value)
        If d < dtS Or d > dtE Then GoTo Skip2
        Dim lbl As String: lbl = DateLabel(d)
        Dim cv As String: cv = Trim(CStr(dws.Cells(r, colC).Value))
        If cv = "" Or cv = "Empty" Then GoTo Skip2
        Dim p() As String: p = Split(cv, Chr(12288))
        Dim big   As String: big   = Trim(p(0))
        Dim small As String: small = IIf(UBound(p) >= 1, Trim(p(1)), "（その他）")
        Dim kbs   As String: kbs   = big & Chr(9) & small
        Dim kf    As String: kf    = kbs & Chr(9) & lbl
        If Not bigSet.Exists(big) Then bigSet.Add big, True: bigOrder.Add big
        If Not smallSet.Exists(kbs) Then smallSet.Add kbs, big: smallOrder.Add kbs
        If Not dateSet.Exists(lbl) Then dateSet.Add lbl, True: dateOrder.Add lbl
        counts(kf) = IIf(counts.Exists(kf), counts(kf) + 1, 1)
Skip2:
    Next r

    StyleTitle ws, "カテゴリー別詳細", dtS, dtE
    ws.Cells(2, 1).Value = "大分類"
    ws.Cells(2, 2).Value = "小分類"
    Dim col As Long: col = 3
    Dim dk As Variant
    For Each dk In dateOrder
        ws.Cells(2, col).Value = dk: col = col + 1
    Next dk
    Dim tcol As Long: tcol = col
    ws.Cells(2, tcol).Value = "合計"
    StyleHeader ws, 2, tcol

    Dim rOut As Long: rOut = 3
    Dim bk As Variant
    For Each bk In bigOrder
        Dim first As Boolean: first = True
        Dim sk As Variant
        For Each sk In smallOrder
            If smallSet(sk) = bk Then
                Dim pp() As String: pp = Split(sk, Chr(9))
                If first Then ws.Cells(rOut, 1).Value = pp(0): first = False
                ws.Cells(rOut, 2).Value = pp(1)
                col = 3
                For Each dk In dateOrder
                    Dim kf2 As String: kf2 = sk & Chr(9) & dk
                    ws.Cells(rOut, col).Value = IIf(counts.Exists(kf2), counts(kf2), 0)
                    col = col + 1
                Next dk
                ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 3).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
                rOut = rOut + 1
            End If
        Next sk
    Next bk

    ws.Cells(rOut, 1).Value = "合計"
    col = 3
    For Each dk In dateOrder
        ws.Cells(rOut, col).Value = "=SUM(" & ws.Cells(3, col).Address(False, False) & ":" & ws.Cells(rOut - 1, col).Address(False, False) & ")"
        col = col + 1
    Next dk
    ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 3).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
    StyleData ws, 3, rOut - 1, 1, tcol
    StyleTotal ws, rOut, 1, tcol

    Dim rr As Long
    For rr = 3 To rOut - 1
        If Trim(ws.Cells(rr, 1).Value) <> "" Then
            ws.Cells(rr, 1).Interior.Color = RGB(214, 228, 247)
            ws.Cells(rr, 1).Font.Bold = True
        End If
    Next rr

    Dim ms As Long: ms = 3
    Dim cb As String: cb = ws.Cells(3, 1).Value
    For rr = 4 To rOut
        Dim cv2 As String: cv2 = Trim(ws.Cells(rr, 1).Value)
        If cv2 <> "" And cv2 <> cb Then
            If rr - 1 > ms Then
                ws.Range(ws.Cells(ms, 1), ws.Cells(rr - 1, 1)).Merge
                ws.Cells(ms, 1).VerticalAlignment = xlCenter
                ws.Cells(ms, 1).HorizontalAlignment = xlCenter
            End If
            ms = rr: cb = cv2
        End If
    Next rr
    AutoFitSheet ws
End Sub

' ════════════════════════════════════════════
' ⑤ カテゴリー別（週比較）
' ════════════════════════════════════════════
Sub BuildCatWeekSheet(dtS As Date, dtE As Date)
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_CAT_WEEK)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    Dim colC As Long: colC = ColIdx(dws, "業務分類詳細")
    If colD = 0 Or colC = 0 Then Exit Sub

    Dim thisMonday As Date: thisMonday = WeekMon(dtE)
    Dim lastMonday As Date: lastMonday = thisMonday - 7
    Dim thisSunday As Date: thisSunday = thisMonday + 6
    Dim lastSunday As Date: lastSunday = lastMonday + 6

    Dim thisCnt As Object: Set thisCnt = CreateObject("Scripting.Dictionary")
    Dim lastCnt As Object: Set lastCnt = CreateObject("Scripting.Dictionary")
    Dim bigOrder As New Collection
    Dim bigSet   As Object: Set bigSet = CreateObject("Scripting.Dictionary")

    Dim r As Long
    For r = 2 To lastRow
        If Not IsValidDate(dws.Cells(r, colD).Value) Then GoTo Skip5
        Dim d As Date: d = ToDate(dws.Cells(r, colD).Value)
        Dim cv As String: cv = Trim(CStr(dws.Cells(r, colC).Value))
        If cv = "" Or cv = "Empty" Then GoTo Skip5
        Dim big As String: big = Trim(Split(cv, Chr(12288))(0))
        If Not bigSet.Exists(big) Then bigSet.Add big, True: bigOrder.Add big
        If d >= thisMonday And d <= thisSunday Then
            thisCnt(big) = IIf(thisCnt.Exists(big), thisCnt(big) + 1, 1)
        End If
        If d >= lastMonday And d <= lastSunday Then
            lastCnt(big) = IIf(lastCnt.Exists(big), lastCnt(big) + 1, 1)
        End If
Skip5:
    Next r

    StyleTitle ws, "カテゴリー別（週比較）", dtS, dtE

    Dim thisLbl As String
    Dim lastLbl As String
    thisLbl = "今週（" & Month(thisMonday) & "/" & Day(thisMonday) & "～" & Month(thisSunday) & "/" & Day(thisSunday) & "）"
    lastLbl = "先週（" & Month(lastMonday) & "/" & Day(lastMonday) & "～" & Month(lastSunday) & "/" & Day(lastSunday) & "）"

    ws.Cells(2, 1).Value = "大分類"
    ws.Cells(2, 2).Value = thisLbl
    ws.Cells(2, 3).Value = lastLbl
    ws.Cells(2, 4).Value = "差分"
    StyleHeader ws, 2, 4
    ws.Rows(2).RowHeight = 36

    Dim rOut As Long: rOut = 3
    Dim bk As Variant
    For Each bk In bigOrder
        Dim tc As Long: tc = IIf(thisCnt.Exists(bk), thisCnt(bk), 0)
        Dim lc As Long: lc = IIf(lastCnt.Exists(bk), lastCnt(bk), 0)
        Dim df As Long: df = tc - lc
        ws.Cells(rOut, 1).Value = bk
        ws.Cells(rOut, 2).Value = tc
        ws.Cells(rOut, 3).Value = lc
        ws.Cells(rOut, 4).Value = df
        ColorDiff ws.Cells(rOut, 4), df
        rOut = rOut + 1
    Next bk

    ws.Cells(rOut, 1).Value = "合計"
    ws.Cells(rOut, 2).Value = "=SUM(" & ws.Cells(3, 2).Address(False, False) & ":" & ws.Cells(rOut - 1, 2).Address(False, False) & ")"
    ws.Cells(rOut, 3).Value = "=SUM(" & ws.Cells(3, 3).Address(False, False) & ":" & ws.Cells(rOut - 1, 3).Address(False, False) & ")"
    ws.Cells(rOut, 4).Value = "=" & ws.Cells(rOut, 2).Address(False, False) & "-" & ws.Cells(rOut, 3).Address(False, False)
    StyleData ws, 3, rOut - 1, 1, 4
    StyleTotal ws, rOut, 1, 4

    rOut = rOut + 2
    ws.Cells(rOut, 1).Value = "【週の判定基準】終了日（" & Format(dtE, "m/d") & "）を含む週を今週とします。差分 = 今週 - 先週"
    With ws.Cells(rOut, 1).Font
        .Name = "Meiryo UI": .Size = 9: .Color = RGB(100, 100, 100): .Italic = True
    End With

    AutoFitSheet ws
    ws.Columns(1).ColumnWidth = 24
    ws.Columns(2).ColumnWidth = 22
    ws.Columns(3).ColumnWidth = 22
    ws.Columns(4).ColumnWidth = 12
End Sub

' ════════════════════════════════════════════
' ⑥ 問い合わせの多い区局
' ════════════════════════════════════════════
Sub BuildBureauSheet(dtS As Date, dtE As Date)
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_BUREAU)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD As Long: colD = ColIdx(dws, "初回受付日時")
    Dim colB As Long: colB = ColIdx(dws, "所属（局）")
    If colD = 0 Or colB = 0 Then Exit Sub

    Dim counts  As Object: Set counts  = CreateObject("Scripting.Dictionary")
    Dim burTot  As Object: Set burTot  = CreateObject("Scripting.Dictionary")
    Dim dateOrder As New Collection
    Dim dateSet   As Object: Set dateSet = CreateObject("Scripting.Dictionary")

    Dim r As Long
    For r = 2 To lastRow
        If Not IsValidDate(dws.Cells(r, colD).Value) Then GoTo Skip6
        Dim d As Date: d = ToDate(dws.Cells(r, colD).Value)
        If d < dtS Or d > dtE Then GoTo Skip6
        Dim lbl As String: lbl = DateLabel(d)
        Dim bv As String: bv = Trim(CStr(dws.Cells(r, colB).Value))
        If bv = "" Or bv = "Empty" Then GoTo Skip6
        If Not dateSet.Exists(lbl) Then dateSet.Add lbl, True: dateOrder.Add lbl
        burTot(bv) = IIf(burTot.Exists(bv), burTot(bv) + 1, 1)
        Dim kk As String: kk = bv & Chr(9) & lbl
        counts(kk) = IIf(counts.Exists(kk), counts(kk) + 1, 1)
Skip6:
    Next r

    StyleTitle ws, "問い合わせの多い区局", dtS, dtE
    ws.Cells(2, 1).Value = "区局名"
    Dim col As Long: col = 2
    Dim dk As Variant
    For Each dk In dateOrder
        ws.Cells(2, col).Value = dk: col = col + 1
    Next dk
    Dim tcol As Long: tcol = col
    ws.Cells(2, tcol).Value = "合計"
    StyleHeader ws, 2, tcol

    Dim bKeys()   As String: ReDim bKeys(burTot.Count - 1)
    Dim bTotals() As Long:   ReDim bTotals(burTot.Count - 1)
    Dim bi As Long: bi = 0
    Dim bk As Variant
    For Each bk In burTot.Keys
        bKeys(bi) = bk: bTotals(bi) = burTot(bk): bi = bi + 1
    Next bk
    Dim i As Long, j As Long
    For i = 0 To UBound(bKeys) - 1
        For j = i + 1 To UBound(bKeys)
            If bTotals(j) > bTotals(i) Then
                Dim ts As String: ts = bKeys(i): bKeys(i) = bKeys(j): bKeys(j) = ts
                Dim tl As Long:   tl = bTotals(i): bTotals(i) = bTotals(j): bTotals(j) = tl
            End If
        Next j
    Next i

    Dim rOut As Long: rOut = 3
    For bi = 0 To UBound(bKeys)
        ws.Cells(rOut, 1).Value = bKeys(bi)
        col = 2
        For Each dk In dateOrder
            Dim kk2 As String: kk2 = bKeys(bi) & Chr(9) & dk
            ws.Cells(rOut, col).Value = IIf(counts.Exists(kk2), counts(kk2), 0)
            col = col + 1
        Next dk
        ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 2).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
        rOut = rOut + 1
    Next bi

    ws.Cells(rOut, 1).Value = "合計"
    col = 2
    For Each dk In dateOrder
        ws.Cells(rOut, col).Value = "=SUM(" & ws.Cells(3, col).Address(False, False) & ":" & ws.Cells(rOut - 1, col).Address(False, False) & ")"
        col = col + 1
    Next dk
    ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 2).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
    StyleData ws, 3, rOut - 1, 1, tcol
    StyleTotal ws, rOut, 1, tcol
    AutoFitSheet ws
End Sub

' ════════════════════════════════════════════
' ⑦ 受付担当者別対応件数
' ════════════════════════════════════════════
Sub BuildStaffSheet(dtS As Date, dtE As Date)
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(S_STAFF)
    ws.Cells.Clear
    Dim dws As Worksheet: Set dws = ThisWorkbook.Sheets(DATA_SHEET)
    Dim lastRow As Long: lastRow = dws.Cells(dws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then Exit Sub

    Dim colD  As Long: colD = ColIdx(dws, "初回受付日時")
    Dim sCols(1 To 5) As Long
    sCols(1) = ColIdx(dws, "受付担当①"): sCols(2) = ColIdx(dws, "受付担当②")
    sCols(3) = ColIdx(dws, "受付担当③"): sCols(4) = ColIdx(dws, "受付担当④")
    sCols(5) = ColIdx(dws, "受付担当⑤")
    If colD = 0 Or sCols(1) = 0 Then Exit Sub

    Dim counts   As Object: Set counts   = CreateObject("Scripting.Dictionary")
    Dim numCnts  As Object: Set numCnts  = CreateObject("Scripting.Dictionary")
    Dim staffOrder As New Collection
    Dim staffSet   As Object: Set staffSet = CreateObject("Scripting.Dictionary")
    Dim dateOrder  As New Collection
    Dim dateSet    As Object: Set dateSet  = CreateObject("Scripting.Dictionary")

    Dim r As Long, sn As Long
    For r = 2 To lastRow
        If Not IsValidDate(dws.Cells(r, colD).Value) Then GoTo Skip7
        Dim d As Date: d = ToDate(dws.Cells(r, colD).Value)
        If d < dtS Or d > dtE Then GoTo Skip7
        Dim lbl As String: lbl = DateLabel(d)
        If Not dateSet.Exists(lbl) Then dateSet.Add lbl, True: dateOrder.Add lbl
        For sn = 1 To 5
            If sCols(sn) = 0 Then GoTo NextSN
            Dim sv As String: sv = Trim(CStr(dws.Cells(r, sCols(sn)).Value))
            If sv = "" Or sv = "Empty" Then GoTo NextSN
            If Not staffSet.Exists(sv) Then staffSet.Add sv, True: staffOrder.Add sv
            Dim kk As String: kk = sv & Chr(9) & lbl
            counts(kk) = IIf(counts.Exists(kk), counts(kk) + 1, 1)
            Dim knk As String: knk = sv & Chr(9) & CStr(sn)
            numCnts(knk) = IIf(numCnts.Exists(knk), numCnts(knk) + 1, 1)
NextSN:
        Next sn
Skip7:
    Next r

    StyleTitle ws, "受付担当者別対応件数", dtS, dtE
    ws.Cells(2, 1).Value = "担当者名"
    ws.Cells(2, 2).Value = "①担当"
    ws.Cells(2, 3).Value = "②担当"
    ws.Cells(2, 4).Value = "③担当"
    ws.Cells(2, 5).Value = "④担当"
    ws.Cells(2, 6).Value = "⑤担当"

    Dim col As Long: col = 7
    Dim dk As Variant
    For Each dk In dateOrder
        ws.Cells(2, col).Value = dk: col = col + 1
    Next dk
    Dim tcol As Long: tcol = col
    ws.Cells(2, tcol).Value = "合計"
    StyleHeader ws, 2, tcol

    Dim numColors(1 To 5) As Long
    numColors(1) = RGB(31, 56, 100):  numColors(2) = RGB(46, 117, 182)
    numColors(3) = RGB(68, 144, 196): numColors(4) = RGB(100, 162, 206)
    numColors(5) = RGB(144, 186, 220)
    Dim si As Long
    For si = 1 To 5
        ws.Cells(2, si + 1).Interior.Color = numColors(si)
    Next si

    Dim rOut As Long: rOut = 3
    Dim sk As Variant
    For Each sk In staffOrder
        ws.Cells(rOut, 1).Value = sk
        For si = 1 To 5
            Dim knk2 As String: knk2 = sk & Chr(9) & CStr(si)
            ws.Cells(rOut, si + 1).Value = IIf(numCnts.Exists(knk2), numCnts(knk2), 0)
        Next si
        col = 7
        For Each dk In dateOrder
            Dim kk2 As String: kk2 = sk & Chr(9) & dk
            ws.Cells(rOut, col).Value = IIf(counts.Exists(kk2), counts(kk2), 0)
            col = col + 1
        Next dk
        ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 7).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
        rOut = rOut + 1
    Next sk

    ws.Cells(rOut, 1).Value = "合計"
    For si = 1 To 5
        ws.Cells(rOut, si + 1).Value = "=SUM(" & ws.Cells(3, si + 1).Address(False, False) & ":" & ws.Cells(rOut - 1, si + 1).Address(False, False) & ")"
    Next si
    col = 7
    For Each dk In dateOrder
        ws.Cells(rOut, col).Value = "=SUM(" & ws.Cells(3, col).Address(False, False) & ":" & ws.Cells(rOut - 1, col).Address(False, False) & ")"
        col = col + 1
    Next dk
    ws.Cells(rOut, tcol).Value = "=SUM(" & ws.Cells(rOut, 7).Address(False, False) & ":" & ws.Cells(rOut, tcol - 1).Address(False, False) & ")"
    StyleData ws, 3, rOut - 1, 1, tcol
    StyleTotal ws, rOut, 1, tcol
    AutoFitSheet ws
End Sub
